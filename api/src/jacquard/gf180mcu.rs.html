4//! GF180MCU standard-cell library support.
5//!
6//! Sibling of [`crate::sky130`]; together with [`crate::gf180mcu_pdk`] this
7//! module provides cell detection, pin direction, and library identification
8//! for GlobalFoundries' open-source 180 nm MCU PDK across both
9//! `gf180mcu_fd_sc_mcu7t5v0` (7-track, 5 V) and `gf180mcu_fd_sc_mcu9t5v0`
10//! (9-track, 5 V) standard-cell libraries.
11//!
12//! Cell-name convention:
13//! `gf180mcu_fd_sc_<track>__<celltype>_<drive>`
14//! e.g. `gf180mcu_fd_sc_mcu7t5v0__nand2_1`.
15//!
16//! IO pads (`gf180mcu_fd_io__*`), primitives (`gf180mcu_fd_pr__*`), and IP
17//! blocks (`gf180mcu_fd_ip_*__*`) all share the `gf180mcu_fd_` prefix and
18//! are detected as GF180MCU here so the netlist parser doesn't reject them;
19//! cell-type extraction is conservative and only strips the standard-cell
20//! prefixes. Decomposition for non-standard-cell families lands in later
21//! phases per `docs/plans/gf180mcu-enablement.md`.
22
23use compact_str::CompactString;
24use netlistdb::{Direction, LeafPinProvider};
25use sverilogparse::SVerilogRange;
26
27// Generated by build.rs from the vendored cell-model submodules.
28// Provides `GF180MCU_PIN_TABLE: &[(&str, &[(&str, Direction)])]`.
29include!(concat!(env!("OUT_DIR"), "/gf180mcu_pins.rs"));
30
31/// All GF180MCU cell instantiations share this prefix.
32const GF180MCU_PREFIX: &str = "gf180mcu_fd_";
33
34const SC_7T_PREFIX: &str = "gf180mcu_fd_sc_mcu7t5v0__";
35const SC_9T_PREFIX: &str = "gf180mcu_fd_sc_mcu9t5v0__";
36
37/// Is `name` any GF180MCU cell — standard cell, IO pad, primitive, or IP?
38pub fn is_gf180mcu_cell(name: &str) -> bool {
39    name.starts_with(GF180MCU_PREFIX)
40}
41
42/// Strip the GF180MCU library prefix and the trailing drive-strength
43/// suffix to recover the base cell type.
44///
45/// `gf180mcu_fd_sc_mcu7t5v0__nand2_1` → `nand2`.
46/// `gf180mcu_fd_sc_mcu9t5v0__inv_20` → `inv`.
47///
48/// Non-standard-cell families (IO, PR, IP) return the suffix after the
49/// library prefix verbatim — decomposition for those lands in later
50/// phases.
51pub fn extract_cell_type(name: &str) -> &str {
52    let stripped = name
53        .strip_prefix(SC_7T_PREFIX)
54        .or_else(|| name.strip_prefix(SC_9T_PREFIX))
55        .unwrap_or(name);
56
57    // Drive strength is always a numeric tail after the final underscore
58    // — matches the SKY130 convention.
59    if let Some(last_underscore) = stripped.rfind('_') {
60        let suffix = &stripped[last_underscore + 1..];
61        if !suffix.is_empty() && suffix.chars().all(|c| c.is_ascii_digit()) {
62            return &stripped[..last_underscore];
63        }
64    }
65
66    stripped
67}

69/// LeafPinProvider for GF180MCU standard cells.
70///
71/// All cell pins are scalar (no bus pins), so `width_of` returns `None`.
72/// Non-standard-cell families (IO pads, primitives, IP) are recognised
73/// by [`is_gf180mcu_cell`] for detection purposes but absent from the
74/// generated pin table — instantiating one here is a panic until later
75/// phases stub in port directions for those families.
76pub struct GF180MCULeafPins;
77
78impl LeafPinProvider for GF180MCULeafPins {
79    fn direction_of(
80        &self,
81        macro_name: &CompactString,
82        pin_name: &CompactString,
83        _pin_idx: Option<isize>,
84    ) -> Direction {
85        let cell_type = extract_cell_type(macro_name);
86        for (ct, pins) in GF180MCU_PIN_TABLE {
87            if *ct == cell_type {
88                for (p, d) in *pins {
89                    if *p == pin_name.as_str() {
90                        return *d;
91                    }
92                }
93                panic!(
94                    "Unknown gf180mcu pin on cell type '{cell_type}': \
95                     macro={macro_name}, pin={pin_name}",
96                );
97            }
98        }
99        panic!(
100            "Unknown gf180mcu cell (no pin-table entry for type '{cell_type}'): \
101             macro={macro_name}",
102        );
103    }
104
105    fn width_of(
106        &self,
107        _macro_name: &CompactString,
108        _pin_name: &CompactString,
109    ) -> Option<SVerilogRange> {
110        None
111    }
112}

118    #[test]
119    fn detects_standard_cells_in_both_variants() {
120        assert!(is_gf180mcu_cell("gf180mcu_fd_sc_mcu7t5v0__nand2_1"));
121        assert!(is_gf180mcu_cell("gf180mcu_fd_sc_mcu9t5v0__nand2_1"));

124    #[test]
125    fn detects_io_pr_and_ip_families() {
126        assert!(is_gf180mcu_cell("gf180mcu_fd_io__bi_24t"));
127        assert!(is_gf180mcu_cell("gf180mcu_fd_pr__nmos_3p3"));
128        assert!(is_gf180mcu_cell("gf180mcu_fd_ip_sram__sram64x16m8wm1"));
129    }

131    #[test]
132    fn rejects_non_gf180mcu_names() {
133        assert!(!is_gf180mcu_cell("sky130_fd_sc_hd__nand2_1"));
134        assert!(!is_gf180mcu_cell("AND2_00_0"));
135        assert!(!is_gf180mcu_cell(""));
136        assert!(!is_gf180mcu_cell("gf180mcu"));

139    #[test]
140    fn extracts_base_cell_type_across_variants() {
141        assert_eq!(extract_cell_type("gf180mcu_fd_sc_mcu7t5v0__inv_1"), "inv");
142        assert_eq!(extract_cell_type("gf180mcu_fd_sc_mcu7t5v0__inv_20"), "inv");
143        assert_eq!(extract_cell_type("gf180mcu_fd_sc_mcu9t5v0__nand2_4"), "nand2");
144        assert_eq!(extract_cell_type("gf180mcu_fd_sc_mcu9t5v0__dffq_1"), "dffq");
145        assert_eq!(
146            extract_cell_type("gf180mcu_fd_sc_mcu7t5v0__aoi222_1"),
147            "aoi222"
148        );
149    }

151    #[test]
152    fn extraction_handles_dff_with_polarity_in_name() {
153        // The 'n' in dffnq / dffnrnq is part of the cell type, not a
154        // drive-strength suffix — the drive number is the trailing _1.
155        assert_eq!(extract_cell_type("gf180mcu_fd_sc_mcu7t5v0__dffnq_1"), "dffnq");
156        assert_eq!(extract_cell_type("gf180mcu_fd_sc_mcu7t5v0__dffnrsnq_2"), "dffnrsnq");
157    }

159    #[test]
160    fn extraction_passes_through_unprefixed_names() {
161        // Unprefixed names still get numeric-suffix stripping.
162        assert_eq!(extract_cell_type("inv_1"), "inv");
163        // Non-numeric tails are left intact — no false trim.
164        assert_eq!(extract_cell_type("not_a_cell"), "not_a_cell");
165        assert_eq!(extract_cell_type("plain"), "plain");
166    }

168    #[test]
169    fn pin_table_covers_full_cell_catalogue() {
170        // 7t and 9t share an identical 69-cell catalogue per the gf180mcu
171        // README; the generated table is deduped by base cell type, so we
172        // expect exactly 69 entries.
173        assert_eq!(GF180MCU_PIN_TABLE.len(), 69, "expected 69 cell types");
174    }

176    #[test]
177    fn pin_provider_returns_expected_directions() {
178        let provider = GF180MCULeafPins;
179        let dir = |cell: &str, pin: &str| {
180            provider.direction_of(
181                &CompactString::from(cell),
182                &CompactString::from(pin),
183                None,
184            )
185        };
186        // Spot-check across cell families.
187        assert_eq!(dir("gf180mcu_fd_sc_mcu7t5v0__inv_1", "I"), Direction::I);
188        assert_eq!(dir("gf180mcu_fd_sc_mcu7t5v0__inv_20", "ZN"), Direction::O);
189        assert_eq!(dir("gf180mcu_fd_sc_mcu9t5v0__nand2_4", "A1"), Direction::I);
190        assert_eq!(dir("gf180mcu_fd_sc_mcu9t5v0__nand2_4", "ZN"), Direction::O);
191        assert_eq!(dir("gf180mcu_fd_sc_mcu7t5v0__dffq_1", "CLK"), Direction::I);
192        assert_eq!(dir("gf180mcu_fd_sc_mcu7t5v0__dffq_1", "D"), Direction::I);
193        assert_eq!(dir("gf180mcu_fd_sc_mcu7t5v0__dffq_1", "Q"), Direction::O);
194        // notifier is in the cell's port list — must be modeled as an input
195        // so synthesis output naming it doesn't reject as unknown.
196        assert_eq!(dir("gf180mcu_fd_sc_mcu7t5v0__dffq_1", "notifier"), Direction::I);
197    }

199    #[test]
200    fn pin_provider_handles_both_variants_identically() {
201        let provider = GF180MCULeafPins;
202        for cell in ["aoi22_1", "dffrnq_1", "mux2_1"] {
203            let pin = "Q"; // not all of these have Q; only DFF does — we'll branch
204            let _ = pin;
205            let pin_in = match cell {
206                "aoi22_1" => "A1",
207                "dffrnq_1" => "CLK",
208                "mux2_1" => "I0",
209                _ => unreachable!(),
210            };
211            let macro_7t = CompactString::from(format!("gf180mcu_fd_sc_mcu7t5v0__{cell}"));
212            let macro_9t = CompactString::from(format!("gf180mcu_fd_sc_mcu9t5v0__{cell}"));
213            let pin_cs = CompactString::from(pin_in);
214            assert_eq!(
215                provider.direction_of(&macro_7t, &pin_cs, None),
216                provider.direction_of(&macro_9t, &pin_cs, None),
217                "7t/9t disagreed on {cell}.{pin_in}",
218            );
219        }
220    }

234    #[test]
235    fn netlist_db_round_trips_a_mixed_7t_9t_design() {
236        // Exercise the full LeafPinProvider contract through NetlistDB —
237        // catches missing pin entries, mis-typed directions, and any
238        // bus-width assumption the parser makes.
239        const VERILOG: &str = r#"
240module tiny(A, B, CLK, Y, Q);
241  input A, B, CLK;
242  output Y, Q;
243  wire mid, q1, notifier;
244  gf180mcu_fd_sc_mcu7t5v0__nand2_1 u1 (.A1(A), .A2(B), .ZN(mid));
245  gf180mcu_fd_sc_mcu9t5v0__inv_1   u2 (.I(mid), .ZN(Y));
246  gf180mcu_fd_sc_mcu7t5v0__dffq_1  u3 (.CLK(CLK), .D(mid), .Q(q1), .notifier(notifier));
247  gf180mcu_fd_sc_mcu9t5v0__inv_1   u4 (.I(q1), .ZN(Q));
248endmodule
249"#;
250        let nl = netlistdb::NetlistDB::from_sverilog_source(VERILOG, Some("tiny"), &GF180MCULeafPins);
251        assert!(nl.is_some(), "NetlistDB construction failed");
252    }
