5//! PDK-neutral behavioral parsing and AIG decomposition primitives.
6//!
7//! The `.functional.v` cell-model files shipped by open-source PDKs
8//! (SKY130, GF180MCU, and others that follow the same OpenLane /
9//! OpenROAD ecosystem conventions) use a small fixed grammar:
10//!
11//! - One `module <name>( <ports> );` declaration.
12//! - `input`/`output` port-direction lines.
13//! - A topologically ordered sequence of Verilog gate-primitive
14//!   instantiations (`not`, `buf`, `and`, `or`, `nand`, `nor`,
15//!   `xor`, `xnor`) plus optional UDP instantiations whose names
16//!   are PDK-specific (e.g. `sky130_fd_sc_hd__udp_*`,
17//!   `UDP_GF018hv5v_mcu_sc7_*`).
18//!
19//! The parser is fully prefix-agnostic; UDP entries are surfaced
20//! verbatim and the PDK-specific decomposition layer routes them
21//! to PDK-specific UDP handlers.
22//!
23//! This module owns the PDK-neutral primitives that both `sky130_pdk`
24//! and `gf180mcu_pdk` share:
25//!
26//! - The parser entry points (`parse_functional_model`, `parse_udp`)
27//!   and their AST types (`BehavioralGate`, `BehavioralModel`,
28//!   `UdpRow`, `UdpModel`) — re-exported from `sky130_pdk` until
29//!   those parse routines are physically relocated.
30//! - The AIG-builder helpers: `WireVal`, `GATE_MARKER`,
31//!   `build_chain_gate`, `build_xor_chain`, `build_udp_aig`,
32//!   `finalize_decomp_result`.
33//! - The `DecompResult` type returned by every PDK's `decompose_*`
34//!   entry point.
35//!
36//! PDK-specific lookup structs (sky130's fixed-field `CellInputs`,
37//! gf180's `HashMap<String, usize>`) deliberately stay in their
38//! respective `*_pdk` modules — they aren't actually shared.
39
40use std::collections::HashMap;

48// ============================================================================
49// DecompResult — the universal output type
50// ============================================================================
51
52/// Result of decomposing a cell into AIG operations.
53///
54/// The decomposition produces a sequence of AND gates that must be built
55/// in order, where later gates can reference earlier ones.
56#[derive(Debug, Clone)]
57pub struct DecompResult {
58    /// Sequence of AND gate operations to build.
59    /// Each entry is (input_a_iv, input_b_iv) where the lower bit is inversion.
60    /// References to earlier gates use negative indices (-1 = first gate output, etc.)
61    pub and_gates: Vec<(i64, i64)>,
62    /// Index of the final output (-1 = first gate, -2 = second gate, etc.)
63    /// Positive values reference original inputs.
64    pub output_idx: i64,
65    /// Whether to invert the final output
66    pub output_inverted: bool,
67}
68
69// ============================================================================
70// WireVal — internal builder state
71// ============================================================================
72
73/// Tagged value for tracking what kind of thing a wire holds during
74/// behavioral-model decomposition. Either an AIG pin (real input or
75/// AND gate we built (by gate index).
76///
77/// `pub(crate)` so sibling PDK modules can construct AIG sub-circuits
78/// through the same primitives.
79#[derive(Clone, Copy, Debug)]
80pub(crate) enum WireVal {
81    /// An AIG pin with inversion bit (aigpin_iv). Bit 0 = inverted.
82    AigPin(usize),
83    /// Constant value
84    Const(bool),
85}
86
87impl WireVal {
88    /// Get the aigpin_iv value, creating const-0 = AigPin(0) convention.
89    pub(crate) fn as_aigpin_iv(self) -> i64 {
90        match self {
91            WireVal::AigPin(iv) => iv as i64,
92            WireVal::Const(false) => 0, // const-0
93            WireVal::Const(true) => 1,  // const-1
94        }
95    }
96
97    /// Invert this wire value.
98    pub(crate) fn inverted(self) -> Self {
99        match self {
100            WireVal::AigPin(iv) => WireVal::AigPin(iv ^ 1),
101            WireVal::Const(v) => WireVal::Const(!v),
102        }
103    }
104}

106// ============================================================================
107// AIG construction helpers (GATE_MARKER encoding)
108// ============================================================================
109
110/// Marker bit to distinguish gate references from pin references.
111/// Gate outputs use bit 30 set. This limits us to ~500M gates (more than enough).
112pub(crate) const GATE_MARKER: usize = 1 << 30;
113
114/// Check if an aigpin_iv value is a gate reference.
115fn is_gate_ref(aigpin_iv: usize) -> bool {
116    aigpin_iv & GATE_MARKER != 0
117}
118
119/// Extract gate index from a gate-reference aigpin_iv.
120fn gate_ref_index(aigpin_iv: usize) -> usize {
121    (aigpin_iv & !GATE_MARKER & !1) >> 1
122}
123
124/// Build an AND/NAND/OR/NOR chain over N inputs.
125///
126/// For AND/NAND: compute AND of all inputs, optionally invert at the end.
127/// For OR/NOR: invert all inputs, AND them, optionally invert at the end.
128///   OR(a,b,c) = NOT(AND(NOT a, NOT b, NOT c))
129///   NOR(a,b,c) = AND(NOT a, NOT b, NOT c)
130pub(crate) fn build_chain_gate(
131    inputs: &[WireVal],
132    invert_inputs: bool,
133    invert_output: bool,
134    and_gates: &mut Vec<(i64, i64)>,
135) -> WireVal {
136    assert!(inputs.len() >= 2, "Gate must have at least 2 inputs");
137
138    let inputs: Vec<WireVal> = if invert_inputs {
139        inputs.iter().map(|v| v.inverted()).collect()
140    } else {
141        inputs.to_vec()
142    };
143
144    // Chain 2-input AND gates
145    let mut accum = inputs[0];
146    for input in &inputs[1..] {
147        let a_ref = accum.as_aigpin_iv();
148        let b_ref = input.as_aigpin_iv();
149        and_gates.push((a_ref, b_ref));
150        let gate_idx = and_gates.len() - 1;
151        accum = WireVal::AigPin(GATE_MARKER | (gate_idx << 1));
152    }
153
154    if invert_output {
155        accum.inverted()
156    } else {
157        accum
158    }
159}

161/// Build a 2-input XOR: A ^ B = !(!( A & !B) & !(!A & B))
162fn build_xor_2(a: WireVal, b: WireVal, and_gates: &mut Vec<(i64, i64)>) -> WireVal {
163    let a_iv = a.as_aigpin_iv();
164    let b_iv = b.as_aigpin_iv();
165    let a_inv_iv = a.inverted().as_aigpin_iv();
166    let b_inv_iv = b.inverted().as_aigpin_iv();
167
168    // gate0: A & !B
169    and_gates.push((a_iv, b_inv_iv));
170    let g0 = and_gates.len() - 1;
171    let g0_val = WireVal::AigPin(GATE_MARKER | (g0 << 1));
172
173    // gate1: !A & B
174    and_gates.push((a_inv_iv, b_iv));
175    let g1 = and_gates.len() - 1;
176    let g1_val = WireVal::AigPin(GATE_MARKER | (g1 << 1));
177
178    // gate2: !(A & !B) & !(!A & B)  -- this is XNOR, inverted gives XOR
179    let g0_inv_iv = g0_val.inverted().as_aigpin_iv();
180    let g1_inv_iv = g1_val.inverted().as_aigpin_iv();
181    and_gates.push((g0_inv_iv, g1_inv_iv));
182    let g2 = and_gates.len() - 1;
183    // XOR = NOT(gate2), so return inverted
184    WireVal::AigPin(GATE_MARKER | (g2 << 1) | 1)
185}
186
187/// Build XOR/XNOR chain for multi-input gates.
188pub(crate) fn build_xor_chain(
189    inputs: &[WireVal],
190    invert_output: bool,
191    and_gates: &mut Vec<(i64, i64)>,
192) -> WireVal {
193    assert!(inputs.len() >= 2);
194
195    let mut accum = inputs[0];
196    for input in &inputs[1..] {
197        accum = build_xor_2(accum, *input, and_gates);
198    }
199
200    if invert_output {
201        accum.inverted()
202    } else {
203        accum
204    }
205}

207/// Build AIG for a UDP instantiation by converting truth table to sum-of-products.
208///
209/// `pub(crate)` so sibling PDK modules (e.g. `gf180mcu_pdk::decompose_with_pdk`)
210/// can route their own UDP gate-type prefixes through the same SOP builder.
211pub(crate) fn build_udp_aig(
212    gate: &BehavioralGate,
213    wires: &HashMap<String, WireVal>,
214    udps: &HashMap<String, UdpModel>,
215    and_gates: &mut Vec<(i64, i64)>,
216) -> WireVal {
217    let udp_name = &gate.gate_type;
218    let udp = udps
219        .get(udp_name)
220        .unwrap_or_else(|| panic!("UDP '{}' not found in loaded models", udp_name));
221
222    // Get input wire values
223    let input_vals: Vec<WireVal> = gate
224        .inputs
225        .iter()
226        .map(|name| {
227            wires
228                .get(name)
229                .copied()
230                .unwrap_or_else(|| panic!("Unknown wire '{}' in UDP '{}'", name, udp_name))
231        })
232        .collect();
233
234    assert_eq!(
235        input_vals.len(),
236        udp.inputs.len(),
237        "UDP '{}' expects {} inputs, got {}",
238        udp_name,
239        udp.inputs.len(),
240        input_vals.len()
241    );
242
243    // Build sum-of-products from truth table rows where output=1
244    // Each row with output=1 becomes a product (AND) term.
245    // Product terms are ORed together.
246    //
247    // For rows where output=0, we don't need to do anything explicitly.
248    // Don't-care (?) inputs are omitted from the product term.
249
250    let one_rows: Vec<&UdpRow> = udp.rows.iter().filter(|r| r.output).collect();
251
252    if one_rows.is_empty() {
253        // Output is always 0
254        return WireVal::Const(false);
255    }
256
257    // Build each product term
258    let mut product_terms: Vec<WireVal> = Vec::new();
259
260    for row in &one_rows {
261        // Collect non-don't-care inputs for this product term
262        let mut term_inputs: Vec<WireVal> = Vec::new();
263        for (i, pattern) in row.inputs.iter().enumerate() {
264            match pattern {
265                Some(true) => term_inputs.push(input_vals[i]),
266                Some(false) => term_inputs.push(input_vals[i].inverted()),
267                None => {} // don't-care - omit from product
268            }
269        }
270
271        if term_inputs.is_empty() {
272            // All inputs are don't-care: output is unconditionally 1
273            return WireVal::Const(true);
274        }
275
276        if term_inputs.len() == 1 {
277            product_terms.push(term_inputs[0]);
278        } else {
279            // Build AND chain for this product term
280            let product = build_chain_gate(&term_inputs, false, false, and_gates);
281            product_terms.push(product);
282        }
283    }
284
285    if product_terms.len() == 1 {
286        return product_terms[0];
287    }
288
289    // OR the product terms: OR(a,b,...) = NOT(AND(NOT a, NOT b, ...))
290    build_chain_gate(&product_terms, true, true, and_gates)
291}

293// ============================================================================
294// DecompResult conversion: GATE_MARKER encoding -> standard negative-index encoding
295// ============================================================================
296
297/// Post-process a DecompResult built with GATE_MARKER encoding to use
298/// standard negative-index encoding for the and_gates references.
299pub(crate) fn finalize_decomp_result(and_gates: Vec<(i64, i64)>, output: WireVal) -> DecompResult {
300    // Convert gate references in and_gates from GATE_MARKER to negative indices
301    let converted_gates: Vec<(i64, i64)> = and_gates
302        .iter()
303        .map(|(a, b)| (convert_ref_to_standard(*a), convert_ref_to_standard(*b)))
304        .collect();
305
306    match output {
307        WireVal::AigPin(iv) if is_gate_ref(iv) => {
308            let gate_idx = gate_ref_index(iv);
309            let inverted = (iv & 1) != 0;
310            DecompResult {
311                and_gates: converted_gates,
312                output_idx: -(gate_idx as i64) - 1,
313                output_inverted: inverted,
314            }
315        }
316        WireVal::AigPin(iv) => {
317            let pin_idx = iv >> 1;
318            let inverted = (iv & 1) != 0;
319            DecompResult {
320                and_gates: converted_gates,
321                output_idx: pin_idx as i64,
322                output_inverted: inverted,
323            }
324        }
325        WireVal::Const(v) => DecompResult {
326            and_gates: converted_gates,
327            output_idx: 0,
328            output_inverted: v,
329        },
330    }
331}
332
333/// Convert a single reference value from GATE_MARKER encoding to standard.
334fn convert_ref_to_standard(ref_val: i64) -> i64 {
335    let uval = ref_val as usize;
336    if is_gate_ref(uval) {
337        let gate_idx = gate_ref_index(uval);
338        let inverted = (uval & 1) != 0;
339        let base = -((gate_idx as i64) * 2 + 1);
340        if inverted {
341            base ^ 1
342        } else {
343            base
344        }
345    } else {
346        ref_val
347    }
348}
