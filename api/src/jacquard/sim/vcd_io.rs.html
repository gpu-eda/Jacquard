19/// Hierarchical name representation in VCD.
20#[derive(PartialEq, Eq, Clone, Debug)]
21pub struct VCDHier {
22    pub cur: CompactString,
23    pub prev: Option<Rc<VCDHier>>,
24}
25
26/// Reverse iterator of a [`VCDHier`], yielding cell names
27/// from the bottom to the top module.
28pub struct VCDHierRevIter<'i>(Option<&'i VCDHier>);
29
30impl<'i> Iterator for VCDHierRevIter<'i> {
31    type Item = &'i CompactString;
32
33    #[inline]
34    fn next(&mut self) -> Option<&'i CompactString> {
35        let name = self.0?;
36        if name.cur.is_empty() {
37            return None;
38        }
39        let ret = &name.cur;
40        self.0 = name.prev.as_ref().map(|a| a.as_ref());
41        Some(ret)
42    }
43}
44
45impl<'i> IntoIterator for &'i VCDHier {
46    type Item = &'i CompactString;
47    type IntoIter = VCDHierRevIter<'i>;
48
49    #[inline]
50    fn into_iter(self) -> VCDHierRevIter<'i> {
51        VCDHierRevIter(Some(self))
52    }
53}
54
55impl Hash for VCDHier {
56    fn hash<H: std::hash::Hasher>(&self, state: &mut H) {
57        for s in self.iter() {
58            s.hash(state);
59        }
60    }
61}
62
63#[allow(dead_code)]
64impl VCDHier {
65    #[inline]
66    pub fn single(cur: CompactString) -> Self {
67        VCDHier { cur, prev: None }
68    }
69
70    #[inline]
71    pub fn empty() -> Self {
72        VCDHier {
73            cur: "".into(),
74            prev: None,
75        }
76    }
77
78    #[inline]
79    pub fn is_empty(&self) -> bool {
80        self.cur.as_str() == "" && self.prev.is_none()
81    }
82
83    #[inline]
84    pub fn iter(&self) -> VCDHierRevIter<'_> {
85        self.into_iter()
86    }
87}

91/// Try to match one component in a scope path.
92/// Returns the remaining scope on success, or None on failure.
93pub fn match_scope_path<'i>(mut scope: &'i str, cur: &str) -> Option<&'i str> {
94    if scope.is_empty() {
95        return Some("");
96    }
97    if scope.starts_with('/') {
98        scope = &scope[1..];
99    }
100    if scope.is_empty() {
101        Some("")
102    } else if scope.starts_with(cur) {
103        if scope.len() == cur.len() {
104            Some("")
105        } else if scope.as_bytes()[cur.len()] == b'/' {
106            Some(&scope[cur.len() + 1..])
107        } else {
108            None
109        }
110    } else {
111        None
112    }
113}
114
115/// Find a scope by its path in the VCD hierarchy.
116pub fn find_top_scope<'i>(items: &'i [ScopeItem], top_scope: &'_ str) -> Option<&'i Scope> {
117    for item in items {
118        if let ScopeItem::Scope(scope) = item {
119            if let Some(s1) = match_scope_path(top_scope, scope.identifier.as_str()) {
120                return match s1 {
121                    "" => Some(scope),
122                    _ => find_top_scope(&scope.children[..], s1),
123                };
124            }
125        }
126    }
127    None
128}

130/// Recursively collect all scope paths from VCD header.
131pub fn collect_all_scopes<'a>(
132    items: &'a [ScopeItem],
133    prefix: &str,
134    scopes: &mut Vec<(String, &'a Scope)>,
135) {
136    for item in items {
137        if let ScopeItem::Scope(scope) = item {
138            let scope_path = if prefix.is_empty() {
139                scope.identifier.to_string()
140            } else {
141                format!("{}/{}", prefix, scope.identifier)
142            };
143            scopes.push((scope_path.clone(), scope));
144            collect_all_scopes(&scope.children[..], &scope_path, scopes);
145        }
146    }
147}

149/// Get required input port names from netlistdb.
150pub fn get_required_input_ports(netlistdb: &NetlistDB) -> HashSet<String> {
151    let mut ports = HashSet::new();
152    for i in netlistdb.cell2pin.iter_set(0) {
153        if netlistdb.pindirect[i] != Direction::I {
154            let port_name = netlistdb.pinnames[i].1.to_string();
155            ports.insert(port_name);
156        }
157    }
158    ports
159}
160
161/// Check if a VCD scope contains all required ports.
162pub fn check_scope_contains_ports(scope: &Scope, required_ports: &HashSet<String>) -> bool {
163    let mut found_ports = HashSet::new();
164    for item in &scope.children {
165        if let ScopeItem::Var(var) = item {
166            found_ports.insert(var.reference.to_string());
167        }
168    }
169
170    for port in required_ports {
171        if !found_ports.contains(port) {
172            return false;
173        }
174    }
175    true
176}

178/// Auto-detect VCD scope containing the DUT.
179pub fn auto_detect_vcd_scope<'i>(
180    items: &'i [ScopeItem],
181    netlistdb: &NetlistDB,
182    top_module_name: &str,
183) -> Option<(String, &'i Scope)> {
184    let required_ports = get_required_input_ports(netlistdb);
185
186    if required_ports.is_empty() {
187        clilog::warn!("No input ports found in design - cannot auto-detect VCD scope");
188        return None;
189    }
190
191    let mut all_scopes = Vec::new();
192    collect_all_scopes(items, "", &mut all_scopes);
193
194    if all_scopes.is_empty() {
195        clilog::warn!("No scopes found in VCD file");
196        return None;
197    }
198
199    clilog::debug!(
200        "Searching for VCD scope containing {} input ports",
201        required_ports.len()
202    );
203    clilog::debug!("Required ports: {:?}", required_ports);
204
205    // Try common DUT scope names first
206    let common_names = ["dut", "uut", "DUT", "UUT", top_module_name];
207    for name in &common_names {
208        for (path, scope) in &all_scopes {
209            if path.ends_with(name) && check_scope_contains_ports(scope, &required_ports) {
210                clilog::info!(
211                    "Auto-detected VCD scope: {} (matched common pattern '{}')",
212                    path,
213                    name
214                );
215                return Some((path.clone(), *scope));
216            }
217        }
218    }
219
220    // Try any scope that contains all required ports
221    for (path, scope) in &all_scopes {
222        if check_scope_contains_ports(scope, &required_ports) {
223            clilog::info!(
224                "Auto-detected VCD scope: {} (contains all required ports)",
225                path
226            );
227            return Some((path.clone(), *scope));
228        }
229    }
230
231    clilog::error!("Could not auto-detect VCD scope. Available scopes:");
232    for (path, _) in &all_scopes {
233        clilog::error!("  - {}", path);
234    }
235    clilog::error!("Please specify scope manually with --input-vcd-scope");
236    None
237}

241/// Result of parsing input VCD: the state vectors and cycle timestamps.
242pub struct ParsedInputVCD {
243    /// Flattened input state vectors, one per cycle + trailing state.
244    pub input_states: Vec<u32>,
245    /// (offset_into_input_states, vcd_timestamp) per cycle.
246    pub offsets_timestamps: Vec<(usize, u64)>,
247}

249/// Resolve the top scope from VCD header, either from user-specified path or auto-detection.
250pub fn resolve_vcd_scope<'i>(
251    header_items: &'i [ScopeItem],
252    input_vcd_scope: Option<&str>,
253    netlistdb: &NetlistDB,
254    top_module: Option<&str>,
255) -> &'i Scope {
256    if let Some(scope_path) = input_vcd_scope {
257        clilog::info!("Using user-specified VCD scope: {}", scope_path);
258        find_top_scope(header_items, scope_path).expect("Specified top scope not found in VCD.")
259    } else {
260        let top_module_name = top_module.unwrap_or("top");
261        clilog::info!("No VCD scope specified - attempting auto-detection");
262
263        match auto_detect_vcd_scope(header_items, netlistdb, top_module_name) {
264            Some((_path, scope)) => scope,
265            None => {
266                panic!(
267                    "Failed to auto-detect VCD scope. Please specify --input-vcd-scope manually."
268                );
269            }
270        }
271    }
272}

274/// Match VCD variables to netlist input ports.
275///
276/// Returns:
277/// - `vcd2inp`: maps (vcd_code, bit_position) → netlist pin ID
278/// - `inp_port_given`: set of netlist pin IDs that were found in VCD
279pub fn match_vcd_inputs(
280    top_scope: &Scope,
281    netlistdb: &NetlistDB,
282) -> (HashMap<(u64, usize), usize>, HashSet<usize>) {
283    use sverilogparse::SVerilogRange;
284    use vcd_ng::ReferenceIndex::*;
285
286    let mut vcd2inp = HashMap::new();
287    let mut inp_port_given = HashSet::new();
288
289    let mut match_one_input = |var: &Var, i: Option<isize>, vcd_pos: usize| {
290        let key = (VCDHier::empty(), var.reference.as_str(), i);
291        if let Some(&id) = netlistdb.pinname2id.get(&key as &dyn GeneralPinName) {
292            // Accept top-level inputs (Direction::O — top-level inputs
293            // "output" onto the netlist's nets) *and* top-level inouts
294            // (Direction::Unknown — NetlistDB doesn't model bidir, so
295            // it tags inout ports as Unknown and AIG construction
296            // treats them as primary inputs). Top-level outputs
297            // (Direction::I) are not stimulated and stay rejected.
298            //
299            // Without accepting Unknown, designs with inout chip-top
300            // ports (e.g. wafer.space chip-top netlists where every
301            // pad is an inout) would silently fail to absorb their
302            // own primary-input stimulus VCD.
303            if !matches!(netlistdb.pindirect[id], Direction::O | Direction::Unknown) {
304                return;
305            }
306            vcd2inp.insert((var.code.0, vcd_pos), id);
307            inp_port_given.insert(id);
308        }
309    };
310
311    for scope_item in &top_scope.children[..] {
312        if let ScopeItem::Var(var) = scope_item {
313            match var.index {
314                None => match var.size {
315                    1 => match_one_input(var, None, 0),
316                    w => {
317                        for (pos, i) in (0..w).rev().enumerate() {
318                            match_one_input(var, Some(i as isize), pos)
319                        }
320                    }
321                },
322                Some(BitSelect(i)) => match_one_input(var, Some(i as isize), 0),
323                Some(Range(a, b)) => {
324                    for (pos, i) in SVerilogRange(a as isize, b as isize).enumerate() {
325                        match_one_input(var, Some(i), pos);
326                    }
327                }
328            }
329        }
330    }
331
332    // Warn about missing primary inputs
333    for i in netlistdb.cell2pin.iter_set(0) {
334        if netlistdb.pindirect[i] != Direction::I && !inp_port_given.contains(&i) {
335            clilog::warn!(
336                GATESIM_VCDI_MISSING_PI,
337                "Primary input port {:?} not present in the VCD input",
338                netlistdb.pinnames[i]
339            );
340        }
341    }
342
343    (vcd2inp, inp_port_given)
344}

346/// Parse input VCD flow into state vectors for simulation.
347pub fn parse_input_vcd(
348    vcdflow: &mut vcd_ng::FastFlow<std::fs::File>,
349    vcd2inp: &HashMap<(u64, usize), usize>,
350    aig: &AIG,
351    script: &FlattenedScriptV1,
352    netlistdb: &NetlistDB,
353    max_clock_edges: Option<usize>,
354) -> ParsedInputVCD {
355    use vcd_ng::{FFValueChange, FastFlowToken};
356
357    let mut state = vec![0; script.reg_io_state_size as usize];
358    let mut vcd_time_last_active = u64::MAX;
359    let mut vcd_time = 0;
360    let mut last_vcd_time_active = true;
361    let mut delayed_bit_changes = HashSet::new();
362
363    let mut input_states = Vec::new();
364    let mut offsets_timestamps = Vec::new();
365
366    while let Some(tok) = vcdflow.next_token().unwrap() {
367        match tok {
368            FastFlowToken::Timestamp(t) => {
369                if t == vcd_time {
370                    continue;
371                }
372                if last_vcd_time_active {
373                    input_states.extend(state.iter().copied());
374                    offsets_timestamps.push((input_states.len(), vcd_time_last_active));
375                    for (_, &(pe, ne)) in &aig.clock_pin2aigpins {
376                        if pe != usize::MAX {
377                            let p = *script.input_map.get(&pe).unwrap();
378                            state[p as usize >> 5] &= !(1 << (p & 31));
379                        }
380                        if ne != usize::MAX {
381                            let p = *script.input_map.get(&ne).unwrap();
382                            state[p as usize >> 5] &= !(1 << (p & 31));
383                        }
384                    }
385                    if let Some(max_clock_edges) = max_clock_edges {
386                        if offsets_timestamps.len() >= max_clock_edges {
387                            clilog::info!("reached --max-clock-edges, stop reading input vcd");
388                            break;
389                        }
390                    }
391                }
392                if last_vcd_time_active {
393                    vcd_time_last_active = vcd_time;
394                }
395                vcd_time = t;
396                last_vcd_time_active = false;
397
398                for pos in std::mem::take(&mut delayed_bit_changes) {
399                    state[(pos >> 5) as usize] ^= 1u32 << (pos & 31);
400                }
401            }
402            FastFlowToken::Value(FFValueChange { id, bits }) => {
403                for (pos, b) in bits.iter().enumerate() {
404                    if let Some(&pin) = vcd2inp.get(&(id.0, pos)) {
405                        let aigpin = aig.pin2aigpin_iv[pin];
406                        assert_eq!(aigpin & 1, 0);
407                        let aigpin = aigpin >> 1;
408                        let pos = match script.input_map.get(&aigpin).copied() {
409                            Some(pos) => pos,
410                            None => {
411                                panic!(
412                                    "input pin {:?} (netlist id {}, aigpin {}) not found in output map.",
413                                    netlistdb.pinnames[pin].dbg_fmt_pin(),
414                                    pin,
415                                    aigpin
416                                );
417                            }
418                        };
419                        let old_value = state[(pos >> 5) as usize] >> (pos & 31) & 1;
420                        if old_value
421                            == match b {
422                                b'1' => 1,
423                                _ => 0,
424                            }
425                        {
426                            continue;
427                        }
428                        if let Some((pe, ne)) = aig.clock_pin2aigpins.get(&pin).copied() {
429                            if pe != usize::MAX && old_value == 0 {
430                                last_vcd_time_active = true;
431                                let p = *script.input_map.get(&pe).unwrap();
432                                state[p as usize >> 5] |= 1 << (p & 31);
433                            }
434                            if ne != usize::MAX && old_value == 1 {
435                                last_vcd_time_active = true;
436                                let p = *script.input_map.get(&ne).unwrap();
437                                state[p as usize >> 5] |= 1 << (p & 31);
438                            }
439                            delayed_bit_changes.insert(pos);
440                        } else {
441                            state[(pos >> 5) as usize] ^= 1u32 << (pos & 31);
442                        }
443                    }
444                }
445            }
446        }
447    }
448    input_states.extend(state.iter().copied());
449    clilog::info!("total number of cycles: {}", offsets_timestamps.len());
450
451    ParsedInputVCD {
452        input_states,
453        offsets_timestamps,
454    }
455}

457// ── X-propagation state buffer helpers ───────────────────────────────────────

459/// Expand value-only state buffer into a doubled buffer with X-mask for GPU dispatch.
460///
461/// The input `value_states` has `(num_cycles + 1)` snapshots of `reg_io_state_size` words.
462/// The output has `(num_cycles + 1)` snapshots of `2 * reg_io_state_size` words:
463/// `[values | xmask]` where X-mask is 0xFFFFFFFF for DFF output positions (unknown)
464/// and 0 for primary input positions (known from VCD).
465pub fn expand_states_for_xprop(value_states: &[u32], script: &FlattenedScriptV1) -> Vec<u32> {
466    let rio = script.reg_io_state_size as usize;
467    let num_snapshots = value_states.len() / rio;
468    assert_eq!(value_states.len(), num_snapshots * rio);
469
470    // Build X-mask template: 0xFFFFFFFF for DFF positions, 0 for inputs
471    let mut xmask_template = vec![0xFFFF_FFFFu32; rio];
472    for &pos in script.input_map.values() {
473        xmask_template[(pos >> 5) as usize] &= !(1u32 << (pos & 31));
474    }
475
476    let mut expanded = Vec::with_capacity(num_snapshots * rio * 2);
477    for snap_i in 0..num_snapshots {
478        // Value portion
479        expanded.extend_from_slice(&value_states[snap_i * rio..(snap_i + 1) * rio]);
480        // X-mask portion
481        expanded.extend_from_slice(&xmask_template);
482    }
483    expanded
484}

486/// Extract value-only states from a doubled GPU output buffer.
487///
488/// Input has `(num_cycles + 1)` snapshots of `2 * reg_io_state_size` words.
489/// Returns separate value and X-mask arrays, each with the same layout as the
490/// original value-only state buffer (indexed by `offsets_timestamps` offsets).
491pub fn split_xprop_states(gpu_states: &[u32], reg_io_state_size: usize) -> (Vec<u32>, Vec<u32>) {
492    let eff = reg_io_state_size * 2;
493    let num_snapshots = gpu_states.len() / eff;
494    assert_eq!(gpu_states.len(), num_snapshots * eff);
495
496    let mut values = Vec::with_capacity(num_snapshots * reg_io_state_size);
497    let mut xmasks = Vec::with_capacity(num_snapshots * reg_io_state_size);
498    for snap_i in 0..num_snapshots {
499        let base = snap_i * eff;
500        values.extend_from_slice(&gpu_states[base..base + reg_io_state_size]);
501        xmasks.extend_from_slice(&gpu_states[base + reg_io_state_size..base + eff]);
502    }
503    (values, xmasks)
504}
505
506/// Expand state buffer to include an arrival section for timing VCD output.
507///
508/// Appends `reg_io_state_size` words of zeros after each cycle's existing
509/// state data (values + optional xmask). This reserves space for the GPU
510/// kernel to write per-output arrival times.
511pub fn expand_states_for_arrivals(states: &[u32], script: &FlattenedScriptV1) -> Vec<u32> {
512    let old_eff = if script.xprop_enabled {
513        script.reg_io_state_size as usize * 2
514    } else {
515        script.reg_io_state_size as usize
516    };
517    let rio = script.reg_io_state_size as usize;
518    let new_eff = old_eff + rio;
519    let num_snapshots = states.len() / old_eff;
520    assert_eq!(states.len(), num_snapshots * old_eff);
521
522    let mut expanded = Vec::with_capacity(num_snapshots * new_eff);
523    for snap_i in 0..num_snapshots {
524        // Copy existing data (values + optional xmask)
525        expanded.extend_from_slice(&states[snap_i * old_eff..(snap_i + 1) * old_eff]);
526        // Append zeroed arrival section
527        expanded.extend(std::iter::repeat_n(0u32, rio));
528    }
529    expanded
530}
531
532/// Extract arrival times from GPU output buffer with timing arrivals enabled.
533///
534/// Returns a vector of u32 arrival times with the same snapshot/offset
535/// structure as the value states. Each u32 contains a u16 arrival time
536/// in picoseconds (stored in the low 16 bits).
537pub fn split_arrival_states(gpu_states: &[u32], script: &FlattenedScriptV1) -> Vec<u32> {
538    let eff = script.effective_state_size() as usize;
539    let rio = script.reg_io_state_size as usize;
540    let arrival_offset = script.arrival_state_offset as usize;
541    let num_snapshots = gpu_states.len() / eff;
542    assert_eq!(gpu_states.len(), num_snapshots * eff);
543
544    let mut arrivals = Vec::with_capacity(num_snapshots * rio);
545    for snap_i in 0..num_snapshots {
546        let base = snap_i * eff + arrival_offset;
547        arrivals.extend_from_slice(&gpu_states[base..base + rio]);
548    }
549    arrivals
550}

560/// Set up output VCD writer: add wire definitions and build output mapping.
561pub fn setup_output_vcd(
562    writer: &mut vcd_ng::Writer<std::io::BufWriter<std::fs::File>>,
563    header: &vcd_ng::Header,
564    output_vcd_scope: Option<&str>,
565    netlistdb: &NetlistDB,
566    aig: &AIG,
567    script: &FlattenedScriptV1,
568) -> OutputVCDMapping {
569    use vcd_ng::SimulationCommand;
570
571    if let Some((ratio, unit)) = header.timescale {
572        writer.timescale(ratio, unit).unwrap();
573    }
574    let output_vcd_scope_str = output_vcd_scope.unwrap_or("gem_top_module");
575    let scope_parts = output_vcd_scope_str.split('/').collect::<Vec<_>>();
576    for &scope in &scope_parts {
577        writer.add_module(scope).unwrap();
578    }
579
580    let mut out2vcd = netlistdb
581        .cell2pin
582        .iter_set(0)
583        .filter_map(|i| {
584            if netlistdb.pindirect[i] == Direction::I {
585                let aigpin = aig.pin2aigpin_iv[i];
586                if matches!(aig.drivers[aigpin >> 1], DriverType::InputPort(_)) {
587                    clilog::info!(
588                        "skipped output for port {} as it is a pass-through of input port.",
589                        netlistdb.pinnames[i].dbg_fmt_pin()
590                    );
591                    return None;
592                }
593                if aigpin <= 1 {
594                    return Some((
595                        aigpin,
596                        u32::MAX,
597                        writer
598                            .add_wire(1, &netlistdb.pinnames[i].dbg_fmt_pin().to_string())
599                            .unwrap(),
600                    ));
601                }
602                Some((
603                    aigpin,
604                    *script.output_map.get(&aigpin).unwrap(),
605                    writer
606                        .add_wire(1, &netlistdb.pinnames[i].dbg_fmt_pin().to_string())
607                        .unwrap(),
608                ))
609            } else {
610                None
611            }
612        })
613        .collect::<Vec<_>>();
614
615    emit_extra_observables(writer, aig, script, &mut out2vcd);
616
617    for _ in 0..scope_parts.len() {
618        writer.upscope().unwrap();
619    }
620    writer.enddefinitions().unwrap();

626/// Append wire definitions for `aig.extra_observable_names` onto an
627/// existing output mapping. Used by both `setup_output_vcd` and
628/// `setup_cosim_output_vcd` so the bidir-pad split (`<port>__out` /
629/// `<port>__oe`) and any future named observables show up identically
630/// in regular sim and cosim VCDs.
631///
632/// Must be called inside the active VCD scope (between `add_module`
633/// and `upscope`/`enddefinitions`) so the wires land in the same
634/// scope as the standard cell-0 outputs.
635fn emit_extra_observables<W: std::io::Write>(
636    writer: &mut vcd_ng::Writer<W>,
637    aig: &AIG,
638    script: &FlattenedScriptV1,
639    out2vcd: &mut Vec<(usize, u32, vcd_ng::IdCode)>,
640) {
641    for (&aigpin, names) in aig.extra_observable_names.iter() {
642        // Resolve the GPU state position. Constant aigpins (0/1) are
643        // emitted with `u32::MAX` (special-cased in write_output_vcd
644        // to use the aigpin value directly). Anything in
645        // `primary_outputs` should also be in `script.output_map`;
646        // skip with a warning if the partitioner dropped it (e.g.
647        // dead-stripped const cone).
648        let output_pos = if aigpin <= 1 {
649            u32::MAX
650        } else if let Some(&pos) = script.output_map.get(&aigpin) {
651            pos
652        } else {
653            clilog::warn!(
654                "extra observable {:?} (aigpin {}) has no output_map entry; \
655                 partitioner may have stripped its cone",
656                names,
657                aigpin
658            );
659            continue;
660        };
661        for name in names {
662            let vid = writer.add_wire(1, name).unwrap();
663            out2vcd.push((aigpin, output_pos, vid));
664        }
665    }
666}

668/// Set up output VCD writer for cosim: uses explicit timescale (1ps) instead of VCD header.
669///
670/// Same output port enumeration as `setup_output_vcd`, but takes timescale directly
671/// (cosim has no input VCD header to copy from).
672pub fn setup_cosim_output_vcd<W: std::io::Write>(
673    writer: &mut vcd_ng::Writer<W>,
674    netlistdb: &NetlistDB,
675    aig: &AIG,
676    script: &FlattenedScriptV1,
677) -> OutputVCDMapping {
678    use vcd_ng::SimulationCommand;
679
680    writer.timescale(1, vcd_ng::TimescaleUnit::PS).unwrap();
681    writer.add_module("top").unwrap();
682
683    let mut out2vcd = netlistdb
684        .cell2pin
685        .iter_set(0)
686        .filter_map(|i| {
687            if netlistdb.pindirect[i] == Direction::I {
688                let aigpin = aig.pin2aigpin_iv[i];
689                if matches!(aig.drivers[aigpin >> 1], DriverType::InputPort(_)) {
690                    return None;
691                }
692                if aigpin <= 1 {
693                    return Some((
694                        aigpin,
695                        u32::MAX,
696                        writer
697                            .add_wire(1, &netlistdb.pinnames[i].dbg_fmt_pin().to_string())
698                            .unwrap(),
699                    ));
700                }
701                Some((
702                    aigpin,
703                    *script.output_map.get(&aigpin).unwrap(),
704                    writer
705                        .add_wire(1, &netlistdb.pinnames[i].dbg_fmt_pin().to_string())
706                        .unwrap(),
707                ))
708            } else {
709                None
710            }
711        })
712        .collect::<Vec<_>>();
713
714    emit_extra_observables(writer, aig, script, &mut out2vcd);
715
716    writer.upscope().unwrap();
717    writer.enddefinitions().unwrap();
718    writer.begin(SimulationCommand::Dumpvars).unwrap();
719
720    clilog::info!(
721        "Cosim output VCD: {} output signals registered (timescale=1ps)",
722        out2vcd.len()
723    );
724
725    OutputVCDMapping { out2vcd }
726}

758fn write_output_vcd_impl(
759    writer: &mut vcd_ng::Writer<std::io::BufWriter<std::fs::File>>,
760    output_mapping: &OutputVCDMapping,
761    offsets_timestamps: &[(usize, u64)],
762    states: &[u32],
763    xmask_states: Option<&[u32]>,
764) {
765    use vcd_ng::Value;
766
767    clilog::info!("write out vcd");
768    // Use 3 to represent X in the last_val tracker (0=V0, 1=V1, 2=initial, 3=X)
769    let mut last_val = vec![2u32; output_mapping.out2vcd.len()];
770    let mut x_output_count = 0u64;
771    for &(offset, timestamp) in offsets_timestamps {
772        if timestamp == u64::MAX {
773            continue;
774        }
775        writer.timestamp(timestamp).unwrap();
776        for (i, &(output_aigpin, output_pos, vid)) in output_mapping.out2vcd.iter().enumerate() {
777            let (value_new, is_x) = match output_pos {
778                u32::MAX => {
779                    assert!(output_aigpin <= 1);
780                    (output_aigpin as u32, false)
781                }
782                output_pos => {
783                    let v = states[offset + (output_pos >> 5) as usize] >> (output_pos & 31) & 1;
784                    let x = xmask_states
785                        .map(|xm| {
786                            xm[offset + (output_pos >> 5) as usize] >> (output_pos & 31) & 1 != 0
787                        })
788                        .unwrap_or(false);
789                    (v, x)
790                }
791            };
792
793            // Encode: 0=V0, 1=V1, 3=X
794            let encoded = if is_x { 3 } else { value_new };
795            if encoded == last_val[i] {
796                continue;
797            }
798            last_val[i] = encoded;
799            if is_x {
800                x_output_count += 1;
801            }
802            writer
803                .change_scalar(
804                    vid,
805                    match (is_x, value_new) {
806                        (true, _) => Value::X,
807                        (false, 1) => Value::V1,
808                        _ => Value::V0,
809                    },
810                )
811                .unwrap();
812        }
813    }
814    if x_output_count > 0 {
815        clilog::warn!(
816            "VCD output contains {} X-value transitions across all signals",
817            x_output_count
818        );
819    }
820}

822/// Write simulation results to output VCD with timing-accurate timestamps.
823///
824/// Instead of placing all output transitions at clock edges, each signal's
825/// transition is offset by its arrival time (in picoseconds) from the GPU
826/// kernel's arrival sideband.
827///
828/// `arrival_states` has the same snapshot/offset layout as `states`.
829/// Each u32 word contains a u16 arrival time in picoseconds (low 16 bits).
830/// `timescale` is the VCD timescale (ratio, unit) for converting ps to VCD time.
831pub fn write_output_vcd_timed<W: std::io::Write>(
832    writer: &mut vcd_ng::Writer<W>,
833    output_mapping: &OutputVCDMapping,
834    offsets_timestamps: &[(usize, u64)],
835    states: &[u32],
836    xmask_states: Option<&[u32]>,
837    arrival_states: &[u32],
838    timescale: Option<(u32, vcd_ng::TimescaleUnit)>,
839) {
840    use vcd_ng::Value;
841
842    clilog::info!("write out timed vcd");
843
844    // Compute how many picoseconds each VCD time unit represents.
845    // E.g., timescale=(1, PS) → 1 ps/unit, (1, NS) → 1000 ps/unit.
846    let ps_per_vcd_unit: f64 = match timescale {
847        Some((ratio, unit)) => {
848            // divisor: units-per-second. PS=1e12, NS=1e9.
849            // ps_per_vcd_unit = ratio * (PS_divisor / unit_divisor)
850            let ps_divisor = vcd_ng::TimescaleUnit::PS.divisor() as f64;
851            let unit_divisor = unit.divisor() as f64;
852            (ratio as f64) * (ps_divisor / unit_divisor)
853        }
854        None => 1.0, // Assume 1ps if no timescale
855    };
856
857    // Collect transitions: (actual_timestamp, output_index, value, is_x)
858    let mut last_val = vec![2u32; output_mapping.out2vcd.len()]; // 2 = initial
859    let mut timed_transitions: Vec<(u64, usize, u32, bool)> = Vec::new();
860    let mut x_output_count = 0u64;
861
862    for &(offset, timestamp) in offsets_timestamps {
863        if timestamp == u64::MAX {
864            continue;
865        }
866
867        for (i, &(output_aigpin, output_pos, _vid)) in output_mapping.out2vcd.iter().enumerate() {
868            let (value_new, is_x) = match output_pos {
869                u32::MAX => {
870                    assert!(output_aigpin <= 1);
871                    (output_aigpin as u32, false)
872                }
873                output_pos => {
874                    let v = states[offset + (output_pos >> 5) as usize] >> (output_pos & 31) & 1;
875                    let x = xmask_states
876                        .map(|xm| {
877                            xm[offset + (output_pos >> 5) as usize] >> (output_pos & 31) & 1 != 0
878                        })
879                        .unwrap_or(false);
880                    (v, x)
881                }
882            };
883
884            let encoded = if is_x { 3 } else { value_new };
885            if encoded == last_val[i] {
886                continue;
887            }
888            last_val[i] = encoded;
889            if is_x {
890                x_output_count += 1;
891            }
892
893            // Get arrival time for this output position.
894            // Arrivals are stored one u32 per word position (same indexing as values).
895            // Each u32 word in the arrival section holds a u16 arrival in the low bits,
896            // shared by all 32 output bits packed in the corresponding value word.
897            let arrival_ps = if output_pos != u32::MAX {
898                let word_idx = (output_pos >> 5) as usize;
899                (arrival_states[offset + word_idx] & 0xFFFF) as u64
900            } else {
901                0u64
902            };
903
904            // Convert arrival_ps to VCD time units and add to base timestamp
905            let arrival_vcd_units = if ps_per_vcd_unit > 0.0 {
906                (arrival_ps as f64 / ps_per_vcd_unit).round() as u64
907            } else {
908                0
909            };
910            let actual_timestamp = timestamp + arrival_vcd_units;
911
912            timed_transitions.push((actual_timestamp, i, value_new, is_x));
913        }
914    }
915
916    // Sort by timestamp (stable sort preserves signal order within same timestamp)
917    timed_transitions.sort_by_key(|&(ts, _, _, _)| ts);
918
919    // Write sorted transitions
920    let mut current_timestamp = u64::MAX;
921    for &(ts, i, value_new, is_x) in &timed_transitions {
922        if ts != current_timestamp {
923            writer.timestamp(ts).unwrap();
924            current_timestamp = ts;
925        }
926        let (_, _, vid) = output_mapping.out2vcd[i];
927        writer
928            .change_scalar(
929                vid,
930                match (is_x, value_new) {
931                    (true, _) => Value::X,
932                    (false, 1) => Value::V1,
933                    _ => Value::V0,
934                },
935            )
936            .unwrap();
937    }
938
939    clilog::info!("Timed VCD: {} transitions written", timed_transitions.len());
940    if x_output_count > 0 {
941        clilog::warn!(
942            "VCD output contains {} X-value transitions across all signals",
943            x_output_count
944        );
945    }
946}

950/// Mapping of primary input signals to VCD variable IDs for stimulus VCD output.
951pub struct StimulusVCDMapping {
952    /// (state_bit_pos, vcd_variable_id, is_clock_pin)
953    pub signals: Vec<(u32, vcd_ng::IdCode, bool)>,
954    /// Clock period in picoseconds (for timestamp generation).
955    pub clock_period_ps: u64,
956}
957
958/// Set up a stimulus VCD writer: register all primary input signals and build mapping.
959///
960/// Iterates all `DriverType::InputPort` entries in the AIG to find every primary
961/// input signal. Clock pins are identified via `aig.clock_pin2aigpins`.
962/// Internal clock flag signals (`DriverType::InputClockFlag`) are excluded since
963/// they are AIG-internal and not real ports.
964pub fn setup_stimulus_vcd(
965    writer: &mut vcd_ng::Writer<std::io::BufWriter<std::fs::File>>,
966    netlistdb: &NetlistDB,
967    aig: &AIG,
968    script: &FlattenedScriptV1,
969    clock_period_ps: u64,
970) -> StimulusVCDMapping {
971    use vcd_ng::SimulationCommand;
972
973    writer.timescale(1, vcd_ng::TimescaleUnit::PS).unwrap();
974    writer.add_module("top").unwrap();
975
976    let mut signals = Vec::new();
977
978    for (aigpin_idx, driv) in aig.drivers.iter().enumerate() {
979        if let DriverType::InputPort(pinid) = driv {
980            if let Some(&pos) = script.input_map.get(&aigpin_idx) {
981                let pn = &netlistdb.pinnames[*pinid];
982                let base_name = pn.pin_type();
983                let index = pn
984                    .bus_id()
985                    .map(|i| vcd_ng::ReferenceIndex::BitSelect(i as i32));
986                let is_clock = aig.clock_pin2aigpins.contains_key(pinid);
987                let vid = writer
988                    .add_var(vcd_ng::VarType::Wire, 1, base_name, index)
989                    .unwrap();
990                signals.push((pos, vid, is_clock));
991                clilog::debug!(
992                    "stimulus VCD: pin '{}' pos={} clock={}",
993                    pn.dbg_fmt_pin(),
994                    pos,
995                    is_clock
996                );
997            }
998        }
999    }
1000
1001    writer.upscope().unwrap();
1002    writer.enddefinitions().unwrap();
1003    writer.begin(SimulationCommand::Dumpvars).unwrap();
1004
1005    // Write initial values (all 0)
1006    for &(_, vid, _) in &signals {
1007        writer.change_scalar(vid, vcd_ng::Value::V0).unwrap();
1008    }
1009    writer.end().unwrap();
1010
1011    clilog::info!(
1012        "Stimulus VCD: {} primary input signals registered (clock_period={}ps)",
1013        signals.len(),
1014        clock_period_ps
1015    );
1016
1017    StimulusVCDMapping {
1018        signals,
1019        clock_period_ps,
1020    }
1021}

1336    fn test_write_output_vcd_timed_basic() {
1337        // Set up a minimal output VCD scenario with known arrivals.
1338        // 2 output signals at word positions 0 and 1.
1339        // Cycle 0 → cycle 1: both flip, with arrival 350ps and 885ps.
1340        // VCD timescale = 1ps, so arrival directly maps to VCD units.
1341        use std::io::BufWriter;
1342
1343        let rio = 2u32;
1344        let _script = make_timing_script(rio, false, true);
1345
1346        // Create output mapping: 2 signals, bit positions 0 and 32
1347        let out2vcd = vec![
1348            (2, 0u32, vcd_ng::IdCode(0)),  // signal A at bit 0 (word 0)
1349            (4, 32u32, vcd_ng::IdCode(1)), // signal B at bit 32 (word 1)
1350        ];
1351        let output_mapping = OutputVCDMapping { out2vcd };
1352
1353        // 2 snapshots: rio=2 words each (just values, no xprop)
1354        // offsets are into value-only buffer
1355        let offsets_timestamps: Vec<(usize, u64)> = vec![
1356            (0, 0),     // cycle 0 at t=0
1357            (2, 10000), // cycle 1 at t=10000ps (clock edge)
1358        ];
1359        // State values: snap 0 all zeros, snap 1 both bits set
1360        let states: Vec<u32> = vec![
1361            0x0000_0000,
1362            0x0000_0000, // snap 0: A=0, B=0
1363            0x0000_0001,
1364            0x0000_0001, // snap 1: A=1, B=1
1365        ];
1366        // Arrival times: snap 0 all zero, snap 1 has 350ps for word 0, 885ps for word 1
1367        let arrival_states: Vec<u32> = vec![
1368            0, 0, // snap 0
1369            350, 885, // snap 1: A arrives at 350ps, B arrives at 885ps
1370        ];
1371
1372        // Write to in-memory buffer
1373        let mut buf = Vec::new();
1374        {
1375            let bufwriter = BufWriter::new(&mut buf);
1376            let mut writer = vcd_ng::Writer::new(bufwriter);
1377
1378            writer.timescale(1, vcd_ng::TimescaleUnit::PS).unwrap();
1379            writer.add_module("test").unwrap();
1380            writer.add_wire(1, "A").unwrap();
1381            writer.add_wire(1, "B").unwrap();
1382            writer.upscope().unwrap();
1383            writer.enddefinitions().unwrap();
1384            writer.begin(vcd_ng::SimulationCommand::Dumpvars).unwrap();
1385
1386            write_output_vcd_timed(
1387                &mut writer,
1388                &output_mapping,
1389                &offsets_timestamps,
1390                &states,
1391                None,
1392                &arrival_states,
1393                Some((1, vcd_ng::TimescaleUnit::PS)),
1394            );
1395            // drop writer+bufwriter to flush
1396        }
1397
1398        let vcd_output = String::from_utf8(buf).unwrap();
1399
1400        // Verify the output contains transitions at the correct offset timestamps:
1401        // A flips at 10000 + 350 = 10350ps
1402        // B flips at 10000 + 885 = 10885ps
1403        // A should come before B (sorted by timestamp)
1404        assert!(
1405            vcd_output.contains("#10350"),
1406            "expected timestamp 10350 for signal A, got:\n{}",
1407            vcd_output
1408        );
1409        assert!(
1410            vcd_output.contains("#10885"),
1411            "expected timestamp 10885 for signal B, got:\n{}",
1412            vcd_output
1413        );
1414        // Initial values at t=0
1415        assert!(vcd_output.contains("#0"), "expected initial timestamp 0");
1416
1417        // Verify ordering: 10350 before 10885 in the output
1418        let pos_a = vcd_output.find("#10350").unwrap();
1419        let pos_b = vcd_output.find("#10885").unwrap();
1420        assert!(
1421            pos_a < pos_b,
1422            "signal A (350ps arrival) should appear before B (885ps arrival)"
1423        );
1424    }
