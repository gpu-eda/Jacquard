2//! User-controlled signal tracing for in-design observability.
3//!
4//! Resolves a list of hierarchical signal names (one per line, file-
5//! supplied) into AIG pins and registers each into
6//! `aig.extra_observable_names`, piggybacking on the same VCD-emission
7//! path that the GF180MCU `bi_24t` bidir-pad split already uses
8//! (`emit_extra_observables` in `src/sim/vcd_io.rs`). Each traced
9//! signal becomes a primary output, gets a state-buffer slot at
10//! partition time, and lands in the output VCD alongside top-level
11//! IO. Works uniformly for sequential (DFF Q) and combinational nets
12//! — anything that has a name in netlistdb's `netname2id` is
13//! traceable.
14//!
15//! Trace list registration happens at AIG construction time (before
16//! partitioning), so the file must be supplied via the
17//! `--trace-signals` CLI flag rather than a runtime env var.
18//!
19//! File format: one signal per line. Blank lines and lines whose
20//! first non-whitespace character is `#` are skipped. Hierarchical
21//! names use `.` as a separator; bit indices use trailing `[N]`.
22//!
23//! ```text
24//! # JTAG-DM internal state
25//! chip_core.dm.haltreq_q[0]
26//! chip_core.dm.haltreq_q[1]
27//!
28//! # Yosys-internal nets — same syntax works
29//! chip_core.sram_u._00147_
30//! ```
31//!
32//! Unresolved names are warned, not fatal — the rest of the trace
33//! list still registers. A trailing summary line reports how many
34//! signals registered vs. how many were dropped, so a bad name list
35//! surfaces clearly at startup.
36

44/// Parsed shape of a single signal name from the trace file.
45/// `hier` is the cell-instance hierarchy (everything before the leaf
46/// segment); `leaf` is the final segment name; `bit` is an optional
47/// trailing `[N]` bus index.
48#[derive(Debug, Clone, PartialEq, Eq)]
49pub struct ParsedSignalName {
50    pub hier: HierName,
51    pub leaf: CompactString,
52    pub bit: Option<isize>,
53}
54
55#[derive(Debug)]
56pub enum TraceError {
57    Io {
58        path: std::path::PathBuf,
59        err: std::io::Error,
60    },
61    InvalidName {
62        line: usize,
63        raw: String,
64        reason: String,
65    },
66}
67
68impl std::fmt::Display for TraceError {
69    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
70        match self {
71            Self::Io { path, err } => write!(f, "trace file {}: {}", path.display(), err),
72            Self::InvalidName { line, raw, reason } => {
73                write!(f, "trace file line {line}: `{raw}` — {reason}")
74            }
75        }
76    }
77}
78
79impl std::error::Error for TraceError {}

81/// Read a trace file, returning the list of non-empty, non-comment
82/// signal name strings (raw, not yet parsed). Caller invokes
83/// `parse_signal_name` to convert each into a `ParsedSignalName`.
84pub fn read_trace_file(path: &Path) -> Result<Vec<String>, TraceError> {
85    let contents = std::fs::read_to_string(path).map_err(|err| TraceError::Io {
86        path: path.to_path_buf(),
87        err,
88    })?;
89    let mut names = Vec::new();
90    for line in contents.lines() {
91        let trimmed = line.trim();
92        if trimmed.is_empty() || trimmed.starts_with('#') {
93            continue;
94        }
95        names.push(trimmed.to_string());
96    }
97    Ok(names)
98}

100/// Parse a signal name into one or more candidate
101/// `ParsedSignalName` shapes for netlistdb lookup. Dotted names are
102/// ambiguous in synthesized netlists: Yosys often emits flattened
103/// designs where the entire hierarchy is collapsed into a single
104/// escaped identifier (`\soc.sram.wb_bus__ack`) with dots as
105/// content rather than separators, while other tools preserve the
106/// structural hierarchy and the dots are real splits. Rather than
107/// guess, try both: candidate-0 is the flat-leaf interpretation
108/// (entire name as leaf, empty hier — matches Yosys-style flattened
109/// nets); candidate-1 is the dot-split interpretation (last segment
110/// as leaf, everything before as hier).
111///
112/// Trailing `[N]` bit index is stripped first and applied to both
113/// candidates. Leading `\` is stripped (Verilog escaped-identifier
114/// syntax); we treat the name without the backslash since netlistdb
115/// stores names canonically.
116///
117/// Examples:
118/// - `"clk"` → one candidate: flat leaf `"clk"`, no hier
119/// - `"top.dm.haltreq_q[0]"` → two candidates:
120///   - flat: leaf `"top.dm.haltreq_q"`, bit `Some(0)` (Yosys-flattened)
121///   - dotted: hier `["top","dm"]`, leaf `"haltreq_q"`, bit `Some(0)`
122/// - `"\soc.bus__addr[3]"` → flat: leaf `"soc.bus__addr"`, bit `Some(3)`
123pub fn parse_signal_name(raw: &str) -> Result<Vec<ParsedSignalName>, String> {
124    let raw = raw.trim();
125    if raw.is_empty() {
126        return Err("empty name".to_string());
127    }
128
129    // Verilog escaped-identifier syntax: leading `\` + trailing
130    // space delimits a name with arbitrary characters. Strip both;
131    // netlistdb stores canonical forms.
132    let raw = raw.strip_prefix('\\').unwrap_or(raw);
133    let raw = raw.trim_end();
134
135    // Split off trailing [N] bit index if present. Brackets only
136    // valid at the very end of the name; intermediate brackets
137    // would be part of a (rare) escaped-identifier and aren't
138    // supported.
139    let (head, bit) = if let Some(open) = raw.rfind('[') {
140        if !raw.ends_with(']') {
141            return Err(format!("unmatched `[` in `{raw}`"));
142        }
143        let idx_str = &raw[open + 1..raw.len() - 1];
144        let bit_val: isize = idx_str
145            .parse()
146            .map_err(|e| format!("bit index `{idx_str}` not an integer: {e}"))?;
147        (&raw[..open], Some(bit_val))
148    } else {
149        (raw, None)
150    };
151
152    if head.is_empty() {
153        return Err("empty leaf segment".to_string());
154    }
155
156    let mut candidates = Vec::new();
157
158    // Candidate 0: flat-leaf with explicit bit index (Yosys-
159    // flattened design with dotted escaped-identifier net names,
160    // where the bus index is a structured field in netlistdb).
161    // Always emit; common for synthesized netlists with real bus
162    // declarations.
163    candidates.push(ParsedSignalName {
164        hier: HierName::empty(),
165        leaf: head.into(),
166        bit,
167    });
168
169    // Candidate 1: flat-leaf where the bracket-index is part of
170    // the leaf name itself (Yosys-flattened design where bus
171    // expansion produced 32 scalar wires named
172    // `\foo.bar[0]`...`\foo.bar[31]`, each registered in
173    // netlistdb as `(empty_hier, "foo.bar[N]", None)`). Emit only
174    // when `bit` was actually stripped — otherwise candidate 0
175    // already covers this shape.
176    if bit.is_some() {
177        candidates.push(ParsedSignalName {
178            hier: HierName::empty(),
179            leaf: raw.into(),
180            bit: None,
181        });
182    }
183
184    // Candidate 2: structural-hierarchy split on `.`. Emit only
185    // when the name contains a dot (otherwise candidate 0 covers
186    // it). Skips empty segments to avoid degenerate matches on
187    // names with leading/trailing/consecutive dots.
188    if head.contains('.') {
189        let mut parts: Vec<&str> = head.split('.').collect();
190        let leaf = parts.pop().unwrap();
191        if !leaf.is_empty() && !parts.iter().any(|p| p.is_empty()) {
192            candidates.push(ParsedSignalName {
193                hier: HierName::from_topdown_hier_iter(parts.iter().copied()),
194                leaf: leaf.into(),
195                bit,
196            });
197        }
198    }
199
200    Ok(candidates)
201}

203/// Look up each parsed signal name in netlistdb, register the
204/// resolved aigpin (iv) into `aig.extra_observable_names` and
205/// `aig.primary_outputs`. Returns `(registered, dropped)`.
206/// Unresolved names emit a `clilog::warn!` and increment `dropped`
207/// but don't abort the registration loop — the rest of the list
208/// still lands. Names whose resolved aigpin is the constant-0/1
209/// (`iv <= 1`) are dropped with a warning since they don't need
210/// runtime observation.
211pub fn register_trace_signals(
212    aig: &mut AIG,
213    netlistdb: &NetlistDB,
214    names: &[String],
215) -> (usize, usize) {
216    let mut registered = 0usize;
217    let mut dropped = 0usize;
218
219    for raw in names {
220        let candidates = match parse_signal_name(raw) {
221            Ok(c) => c,
222            Err(e) => {
223                clilog::warn!("--trace-signals: `{raw}` — parse error: {e}");
224                dropped += 1;
225                continue;
226            }
227        };
228
229        // Try each candidate against netname2id; the first hit
230        // wins. Both flat-leaf and dotted-hierarchy shapes are
231        // attempted so the same syntax works for Yosys-flattened
232        // and structurally-hierarchical netlists.
233        let netid = candidates
234            .iter()
235            .find_map(|c| {
236                let key = (c.hier.clone(), c.leaf.clone(), c.bit);
237                netlistdb.netname2id.get(&key).copied()
238            });
239        let Some(netid) = netid else {
240            clilog::warn!(
241                "--trace-signals: `{raw}` — not found in netlistdb \
242                 (tried {} candidate(s))",
243                candidates.len()
244            );
245            dropped += 1;
246            continue;
247        };
248
249        // Any pin on the net carries the same aigpin iv (all pins
250        // share the net's driver). Take the first.
251        let Some(pinid) = netlistdb.net2pin.iter_set(netid).next() else {
252            clilog::warn!("--trace-signals: `{raw}` — net has no pins");
253            dropped += 1;
254            continue;
255        };
256
257        let iv = aig.pin2aigpin_iv[pinid];
258        if iv == usize::MAX {
259            clilog::warn!(
260                "--trace-signals: `{raw}` — pin {} has no AIG mapping \
261                 (likely a non-logic pin or stripped cone)",
262                pinid
263            );
264            dropped += 1;
265            continue;
266        }
267        if iv <= 1 {
268            clilog::warn!(
269                "--trace-signals: `{raw}` — resolves to constant {} \
270                 (iv={}); skipping runtime observation",
271                if iv == 0 { 0 } else { 1 },
272                iv
273            );
274            dropped += 1;
275            continue;
276        }
277
278        aig.primary_outputs.insert(iv);
279        aig.extra_observable_names
280            .entry(iv)
281            .or_default()
282            .push(raw.clone());
283        registered += 1;
284    }
285
286    (registered, dropped)
287}

293    #[test]
294    fn parses_flat_name_without_hier() {
295        let cs = parse_signal_name("clk").unwrap();
296        // No dots — only one candidate (flat).
297        assert_eq!(cs.len(), 1);
298        assert!(cs[0].hier.is_empty());
299        assert_eq!(cs[0].leaf.as_str(), "clk");
300        assert_eq!(cs[0].bit, None);
301    }

303    #[test]
304    fn parses_hierarchical_name_with_bit_emits_all_three_candidates() {
305        let cs = parse_signal_name("top.dm.haltreq_q[0]").unwrap();
306        // Dots + bit index → three candidates.
307        assert_eq!(cs.len(), 3);
308
309        // Candidate 0: flat-leaf, bit as structured field.
310        assert!(cs[0].hier.is_empty());
311        assert_eq!(cs[0].leaf.as_str(), "top.dm.haltreq_q");
312        assert_eq!(cs[0].bit, Some(0));
313
314        // Candidate 1: flat-leaf, bracket in name (Yosys scalar
315        // expansion per bit).
316        assert!(cs[1].hier.is_empty());
317        assert_eq!(cs[1].leaf.as_str(), "top.dm.haltreq_q[0]");
318        assert_eq!(cs[1].bit, None);
319
320        // Candidate 2: dot-split (structurally-hierarchical style).
321        let hier_parts: Vec<&CompactString> = cs[2].hier.iter().collect();
322        assert_eq!(hier_parts.len(), 2);
323        assert_eq!(hier_parts[0].as_str(), "dm"); // bottom-up
324        assert_eq!(hier_parts[1].as_str(), "top");
325        assert_eq!(cs[2].leaf.as_str(), "haltreq_q");
326        assert_eq!(cs[2].bit, Some(0));
327    }

383    /// End-to-end: build a small AIG, register a trace signal by
384    /// hierarchical name, verify the resolved aigpin lands in
385    /// `extra_observable_names` AND `primary_outputs` so the
386    /// existing VCD emission path picks it up.
387    #[test]
388    fn register_resolves_and_pushes_into_extra_observables() {
389        const TOP_VERILOG: &str = "\
390module top(clk, d, q);
391  input clk, d;
392  output q;
393  wire q_internal;
394  DFF u_dff_a(.D(d), .CLK(clk), .Q(q_internal));
395  DFF u_dff_b(.D(q_internal), .CLK(clk), .Q(q));
396endmodule
397";
398        let nl = netlistdb::NetlistDB::from_sverilog_source(
399            TOP_VERILOG,
400            Some("top"),
401            &crate::aigpdk::AIGPDKLeafPins(),
402        )
403        .expect("netlist parse");
404        let mut aig = crate::aig::AIG::from_netlistdb(&nl);
405
406        // The intermediate wire `q_internal` should resolve to the
407        // first DFF's Q output aigpin.
408        let names = vec!["q_internal".to_string()];
409        let (registered, dropped) = register_trace_signals(&mut aig, &nl, &names);
410        assert_eq!(registered, 1);
411        assert_eq!(dropped, 0);
412
413        // The aigpin should now be in extra_observable_names AND
414        // primary_outputs — both are needed for the partitioner to
415        // give it a state-buffer slot.
416        assert_eq!(aig.extra_observable_names.len(), 1);
417        let (iv, names) = aig.extra_observable_names.iter().next().unwrap();
418        assert_eq!(names, &vec!["q_internal".to_string()]);
419        assert!(
420            aig.primary_outputs.contains(iv),
421            "registered trace signal must be in primary_outputs"
422        );
423    }
