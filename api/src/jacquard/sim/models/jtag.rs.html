3//! JTAG replay peripheral model — consumes a recorded
4//! `remote_bitbang` byte stream (the protocol OpenOCD uses to drive
5//! an external JTAG bridge) and drives the design's TCK/TMS/TDI/TRST
6//! input pins from it.
7//!
8//! Stage 1 of the JTAG cosim plan from
9//! [discussion #77](https://github.com/ChipFlow/Jacquard/discussions/77):
10//! deterministic file-replay. The recorded stream is assumed valid
11//! and the design's TDO output is not read back — that lands in
12//! stage 2 alongside the TCP-listener mode where TDO has to be
13//! routed back over the socket to OpenOCD.
14//!
15//! ## `remote_bitbang` byte alphabet
16//!
17//! Per [OpenOCD's `remote_bitbang.c`][openocd-rb]:
18//!
19//! | Byte           | Meaning                                |
20//! |----------------|----------------------------------------|
21//! | `'0'`..=`'7'`  | Set `(TCK, TMS, TDI)` from low 3 bits  |
22//! | `'r'`          | TRST off, SRST off                     |
23//! | `'s'`          | TRST off, SRST on                      |
24//! | `'t'`          | TRST on,  SRST off                     |
25//! | `'u'`          | TRST on,  SRST on                      |
26//! | `'R'`          | Read TDO (ignored in stage 1)          |
27//! | `'B'` / `'b'`  | Blink on / off (no-op)                 |
28//! | `'Q'`          | Quit (treated as end-of-stream)        |
29//!
30//! [openocd-rb]: https://github.com/openocd-org/openocd/blob/master/src/jtag/drivers/remote_bitbang.c
31//!
32//! ## Pacing
33//!
34//! OpenOCD has no concept of simulation time. Stream bytes are
35//! consumed at chip-clock pace: each byte's drive values are held
36//! on the design's input pins for `hold_edges` cosim edges before
37//! the next byte is consumed. Default `hold_edges = 4` — chosen
38//! against the "chip-clock ≥ 2× TCK" assumption that holds by
39//! construction in any chipflow design running a debug TAP.
40//! Configurable per-cosim via `--jtag-hold-cycles <N>` for the
41//! corner case.

46/// Optional pin position + polarity for the design's TRST input.
47///
48/// `active_low: true` (the common case for RV32-Debug TAPs) drives
49/// the configured position LOW when the stream says "TRST on".
50#[derive(Debug, Clone, Copy)]
51pub struct TrstPin {
52    pub position: u32,
53    pub active_low: bool,
54}

56/// CPU-side state for one JTAG replay peripheral.
57pub struct JtagReplayModel {
58    /// Peripheral name (e.g. `jtag_0`).
59    name: String,
60    /// State-buffer position for the design's TCK input.
61    tck_pos: u32,
62    /// State-buffer position for the design's TMS input.
63    tms_pos: u32,
64    /// State-buffer position for the design's TDI input.
65    tdi_pos: u32,
66    /// Optional TRST input. Some chips reset the TAP via a five-cycle
67    /// TMS=1 sequence instead of a dedicated pin.
68    trst: Option<TrstPin>,
69    /// Returned by `driven_positions()`. Built once at construction
70    /// since `PeripheralModel::driven_positions` returns `&[u32]`.
71    driven_positions_arr: Vec<u32>,
72    /// Cosim edges between consuming successive stream bytes.
73    hold_edges: u32,
74    /// Pre-recorded byte stream.
75    bytes: Vec<u8>,
76    /// Next byte to consume.
77    cursor: usize,
78    /// Cosim edges spent on the current drive. Resets to 0 when a
79    /// new byte is consumed.
80    edges_held: u32,
81    /// Currently-driven TCK value (0 / 1).
82    tck: u8,
83    /// Currently-driven TMS value (0 / 1).
84    tms: u8,
85    /// Currently-driven TDI value (0 / 1).
86    tdi: u8,
87    /// Currently-driven raw TRST value (0 / 1, pre-polarity). Combined
88    /// with `trst.active_low` when contributing the override.
89    trst_active: bool,
90    /// Total `'R'` (TDO-read) commands seen in the stream — emitted
91    /// once at end-of-replay so the operator sees that stage 2 would
92    /// have responded to them.
93    tdo_read_requests: u64,
94    /// Total `'s'` / `'t'` / `'u'` (SRST-toggling) commands — flagged
95    /// at end-of-replay since SRST isn't modeled.
96    srst_toggles: u64,
97    /// Set true when the stream's `'Q'` (quit) command is reached.
98    quit: bool,
99}

101impl JtagReplayModel {
102    /// Build a replay model. `name` matches the configured
103    /// peripheral name (typically `jtag_0`). `bytes` is the recorded
104    /// `remote_bitbang` stream — captured via `scripts/capture_bitbang.py`
105    /// or hand-authored for synthetic tests.
106    pub fn new(
107        name: String,
108        tck_pos: u32,
109        tms_pos: u32,
110        tdi_pos: u32,
111        trst: Option<TrstPin>,
112        hold_edges: u32,
113        bytes: Vec<u8>,
114    ) -> Self {
115        assert!(
116            hold_edges > 0,
117            "jtag {name}: hold_edges must be > 0 (got 0)"
118        );
119        let mut driven = vec![tck_pos, tms_pos, tdi_pos];
120        if let Some(t) = trst {
121            driven.push(t.position);
122        }
123        Self {
124            name,
125            tck_pos,
126            tms_pos,
127            tdi_pos,
128            trst,
129            driven_positions_arr: driven,
130            hold_edges,
131            bytes,
132            cursor: 0,
133            // Pre-loaded so the first `step_edge` immediately
134            // consumes the stream's first byte. Subsequent bytes
135            // are spaced `hold_edges` apart.
136            edges_held: hold_edges,
137            // Idle drive at startup: TCK=0, TMS=1 (target Run-Test-Idle
138            // entry), TDI=0, TRST deasserted.
139            tck: 0,
140            tms: 1,
141            tdi: 0,
142            trst_active: false,
143            tdo_read_requests: 0,
144            srst_toggles: 0,
145            quit: false,
146        }
147    }
148
149    /// Consume one byte from the stream, updating the driven values.
150    /// Public for unit testing the parser in isolation.
151    pub fn consume_byte(&mut self, byte: u8) {
152        match byte {
153            // (TCK, TMS, TDI) packed in low 3 bits. Bit assignment per
154            // OpenOCD's remote_bitbang.c: TDI=bit0, TMS=bit1, TCK=bit2.
155            b'0'..=b'7' => {
156                let b = byte - b'0';
157                self.tdi = b & 0b001;
158                self.tms = (b & 0b010) >> 1;
159                self.tck = (b & 0b100) >> 2;
160            }
161            // TRST off (deasserted).
162            b'r' => self.trst_active = false,
163            // TRST off, SRST on — SRST not modeled, count it.
164            b's' => {
165                self.trst_active = false;
166                self.srst_toggles += 1;
167            }
168            // TRST on (asserted), SRST off.
169            b't' => self.trst_active = true,
170            // TRST on, SRST on.
171            b'u' => {
172                self.trst_active = true;
173                self.srst_toggles += 1;
174            }
175            // TDO read — silently counted, response not sent (stage 1
176            // is deterministic replay; stage 2 sends back to socket).
177            b'R' => self.tdo_read_requests += 1,
178            // Blink LED commands — no-op.
179            b'B' | b'b' => {}
180            // Quit — treat as end-of-stream.
181            b'Q' => self.quit = true,
182            // Unknown byte — ignore but warn so corrupted streams
183            // surface during capture/replay. Count too noisy bytes
184            // up-front would require a HashMap; one-shot at end-of-
185            // replay is plenty.
186            other => clilog::warn!(
187                "jtag {}: unknown remote_bitbang byte 0x{:02X} ignored",
188                self.name,
189                other
190            ),
191        }
192    }
193
194    /// True after consuming the stream's final `'Q'` byte or running
195    /// off the end. Cosim doesn't gate on this — replay just goes
196    /// idle on its own.
197    pub fn finished(&self) -> bool {
198        self.quit || self.cursor >= self.bytes.len()
199    }
200
201    /// Diagnostics counters surfaced at end-of-run.
202    pub fn diagnostics(&self) -> JtagDiagnostics {
203        JtagDiagnostics {
204            bytes_consumed: self.cursor,
205            tdo_read_requests: self.tdo_read_requests,
206            srst_toggles: self.srst_toggles,
207            quit: self.quit,
208        }
209    }
210
211    /// Advance one cosim edge. Drains the next byte from the stream
212    /// when the current drive has been held for `hold_edges`. Public
213    /// for unit testing; the `PeripheralModel` impl also calls it.
214    pub fn step_edge(&mut self) {
215        if self.finished() {
216            return;
217        }
218        if self.edges_held >= self.hold_edges {
219            // Time to consume the next byte. Reset to 1 because the
220            // consuming edge itself counts as the first edge of the
221            // new drive's hold window — so a `hold_edges = N` config
222            // gives exactly N cosim edges per byte.
223            let byte = self.bytes[self.cursor];
224            self.cursor += 1;
225            self.consume_byte(byte);
226            self.edges_held = 1;
227        } else {
228            self.edges_held += 1;
229        }
230    }
231}

233/// Counters surfaced when the run finishes — useful for confirming
234/// the replay landed where expected.
235#[derive(Debug, Clone, Copy)]
236pub struct JtagDiagnostics {
237    pub bytes_consumed: usize,
238    pub tdo_read_requests: u64,
239    pub srst_toggles: u64,
240    pub quit: bool,
241}

243impl PeripheralModel for JtagReplayModel {
244    fn name(&self) -> &str {
245        &self.name
246    }
247
248    fn driven_positions(&self) -> &[u32] {
249        &self.driven_positions_arr
250    }
251
252    fn apply_action(&mut self, action: &QueuedAction) {
253        // The replay model is byte-stream driven from the file, not
254        // input.json. Any action targeted at the jtag peripheral is
255        // a config mistake worth warning about.
256        warn_unhandled(&format!("jtag `{}`", self.name), action);
257    }
258
259    fn step_edge(
260        &mut self,
261        _output_state: &[u32],
262        overrides: &mut ModelOverrides,
263        _emitted: &mut Vec<EmittedEvent>,
264    ) {
265        JtagReplayModel::step_edge(self);
266        self.contribute_overrides(overrides);
267    }
268
269    fn contribute_overrides(&self, overrides: &mut ModelOverrides) {
270        overrides.insert(self.tck_pos, self.tck);
271        overrides.insert(self.tms_pos, self.tms);
272        overrides.insert(self.tdi_pos, self.tdi);
273        if let Some(trst) = self.trst {
274            let driven = if trst.active_low {
275                if self.trst_active {
276                    0
277                } else {
278                    1
279                }
280            } else if self.trst_active {
281                1
282            } else {
283                0
284            };
285            overrides.insert(trst.position, driven);
286        }
287    }
288
289    fn is_active(&self) -> bool {
290        !self.finished()
291    }
292}

302    #[test]
303    fn consume_byte_decodes_tck_tms_tdi_packing() {
304        // '0' = 0b000 → all zero. '7' = 0b111 → all one.
305        // Bit assignment per OpenOCD: TDI=bit0, TMS=bit1, TCK=bit2.
306        let mut m = JtagReplayModel::new(
307            "jtag_0".into(),
308            10, 11, 12, None,
309            4,
310            vec![],
311        );
312        m.consume_byte(b'0');
313        assert_eq!(driven(&m), (0, 0, 0));
314        m.consume_byte(b'5'); // 0b101 → TCK=1, TMS=0, TDI=1
315        assert_eq!(driven(&m), (1, 0, 1));
316        m.consume_byte(b'7'); // 0b111
317        assert_eq!(driven(&m), (1, 1, 1));
318        m.consume_byte(b'2'); // 0b010 → TCK=0, TMS=1, TDI=0
319        assert_eq!(driven(&m), (0, 1, 0));
320    }

322    #[test]
323    fn consume_byte_handles_trst_codes() {
324        let mut m = JtagReplayModel::new(
325            "jtag_0".into(),
326            10, 11, 12,
327            Some(TrstPin { position: 13, active_low: true }),
328            4,
329            vec![],
330        );
331        m.consume_byte(b'r');
332        assert!(!m.trst_active);
333        m.consume_byte(b't');

341    }

375    #[test]
376    fn step_edge_holds_then_advances() {
377        // hold_edges=2 with the stream "07": first byte drives all
378        // zeros, after 2 step_edges we advance to '7' which drives
379        // all ones.
380        let mut m = JtagReplayModel::new(
381            "jtag_0".into(),
382            10, 11, 12, None,
383            2,
384            b"07".to_vec(),
385        );
386        // edges_held=0 < hold (2): first call eats byte '0' immediately.
387        m.step_edge();
388        assert_eq!(driven(&m), (0, 0, 0));
389        // edges_held=0 → 1
390        m.step_edge();
391        assert_eq!(driven(&m), (0, 0, 0));
392        // edges_held=1 → 2 → 0 (advance to '7')
393        m.step_edge();
394        assert_eq!(driven(&m), (1, 1, 1));
395        // Stream exhausted after two bytes consumed and hold-out edges.
396        m.step_edge();
397        m.step_edge();
398        assert!(m.finished());
399    }

401    #[test]
402    fn step_edge_stops_at_quit() {
403        let mut m = JtagReplayModel::new(
404            "jtag_0".into(),
405            10, 11, 12, None,
406            1,
407            b"05Q07".to_vec(),
408        );

410            m.step_edge();
411        }
412        assert!(m.finished());
413        // Should have stopped before consuming the post-Q bytes.
414        assert!(m.cursor <= 3, "stopped on Q, cursor={}", m.cursor);

417    #[test]
418    fn contribute_overrides_applies_trst_polarity() {
419        let mut m = JtagReplayModel::new(
420            "jtag_0".into(),
421            10, 11, 12,
422            Some(TrstPin { position: 13, active_low: true }),
423            4,
424            vec![],
425        );
426        m.consume_byte(b't'); // trst on (asserted)
427        let mut overrides = ModelOverrides::new();
428        m.contribute_overrides(&mut overrides);
429        // active_low + asserted → drive LOW
430        assert_eq!(overrides.get(&13), Some(&0));
431
432        m.consume_byte(b'r'); // trst off
433        overrides.clear();
434        m.contribute_overrides(&mut overrides);
435        // active_low + deasserted → drive HIGH
436        assert_eq!(overrides.get(&13), Some(&1));
437    }
