5//! PDK-shared types and dispatch surfaces.
6//!
7//! Owns types and dispatch that span multiple PDKs and don't belong
8//! inside any one PDK's module:
9//!
10//! - [`CellLibrary`] — the detected cell-library variant for a netlist
11//!   (`AIGPDK`, `SKY130`, `GF180MCU`, or `Mixed`).
12//! - [`detect_library`] / [`detect_library_from_file`] — dispatch
13//!   helpers that classify a netlist by the cell prefixes it contains.
14//! - [`PdkVariant`] — the dispatch enum used by `aig.rs` to collapse
15//!   `if is_sky130_cell { sky130_X } else if is_gf180mcu_cell { gf180mcu_X }`
16//!   shaped call sites into a single match.
17//!
18//! Per-PDK predicates (`is_sky130_cell`, `is_aigpdk_cell`,
19//! `is_gf180mcu_cell`) still live with their respective PDK modules;
20//! the detection routines here aggregate them into a single verdict.
21//!
22//! # Why enum dispatch (not `Box<dyn PdkHooks>`)?
23//!
24//! The AIG builder runs these classifiers and accessors on every cell
25//! in a netlist — hundreds of thousands per design, millions for large
26//! Gemmini/NVDLA dumps — and the closed set of PDKs is small (currently
27//! two). A `Box<dyn>` would add a vtable dereference per call and
28//! defeat inlining of the trivial predicates. An enum-with-match keeps
29//! the hot path branch-only and lets the compiler inline the `matches!`
30//! arms.
31//!
32//! Adding a third PDK is a one-arm-per-method expansion, which is
33//! fine for this scale. If the PDK count ever explodes (5+ libraries)
34//! it'd be worth revisiting; until then the simpler shape wins.
35
36use crate::gf180mcu::is_gf180mcu_cell;
37use crate::sky130::{is_aigpdk_cell, is_sky130_cell};

39/// The detected cell library type.
40#[derive(Debug, Clone, Copy, PartialEq, Eq)]
41pub enum CellLibrary {
42    /// AIGPDK standard cells
43    AIGPDK,
44    /// SKY130 open-source PDK
45    SKY130,
46    /// GF180MCU open-source PDK (7t5v0 and/or 9t5v0 standard cell libraries)
47    GF180MCU,
48    /// Cells from more than one library coexist in the netlist (error case).
49    Mixed,
50}
51
52impl std::fmt::Display for CellLibrary {
53    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
54        match self {
55            CellLibrary::AIGPDK => write!(f, "AIGPDK"),
56            CellLibrary::SKY130 => write!(f, "SKY130"),
57            CellLibrary::GF180MCU => write!(f, "GF180MCU"),
58            CellLibrary::Mixed => write!(f, "Mixed"),
59        }
60    }
61}
62
63/// Resolve detection flags into a single library verdict. `Mixed`
64/// fires whenever 2+ libraries are present; absent any recognised
65/// cells, `AIGPDK` is the default (consistent with prior two-library
66/// behaviour).
67fn resolve_library(has_aigpdk: bool, has_sky130: bool, has_gf180mcu: bool) -> CellLibrary {
68    match (has_aigpdk, has_sky130, has_gf180mcu) {
69        (true, true, _) | (true, _, true) | (_, true, true) => CellLibrary::Mixed,
70        (_, true, _) => CellLibrary::SKY130,
71        (_, _, true) => CellLibrary::GF180MCU,
72        (_, false, false) => CellLibrary::AIGPDK,
73    }
74}

76/// Detect which cell library is used in a netlist.
77pub fn detect_library(cell_types: impl Iterator<Item = impl AsRef<str>>) -> CellLibrary {
78    let mut has_aigpdk = false;
79    let mut has_sky130 = false;
80    let mut has_gf180mcu = false;
81
82    for cell_type in cell_types {
83        let name = cell_type.as_ref();
84        if is_sky130_cell(name) {
85            has_sky130 = true;
86        } else if is_gf180mcu_cell(name) {
87            has_gf180mcu = true;
88        } else if is_aigpdk_cell(name) {
89            has_aigpdk = true;
90        }
91        if (has_aigpdk as u8 + has_sky130 as u8 + has_gf180mcu as u8) >= 2 {
92            return CellLibrary::Mixed;
93        }
94    }
95
96    resolve_library(has_aigpdk, has_sky130, has_gf180mcu)
97}

99/// Detect library from a Verilog file by scanning for cell types.
100///
101/// This does a quick scan of the file without full parsing.
102pub fn detect_library_from_file(path: &std::path::Path) -> std::io::Result<CellLibrary> {
103    use std::fs::File;
104    use std::io::{BufRead, BufReader};
105
106    let file = File::open(path)?;
107    let reader = BufReader::new(file);
108
109    let mut has_aigpdk = false;
110    let mut has_sky130 = false;
111    let mut has_gf180mcu = false;
112
113    for line in reader.lines() {
114        let line = line?;
115        if line.contains("sky130_fd_sc_") {
116            has_sky130 = true;
117        }
118        if line.contains("gf180mcu_fd_") {
119            has_gf180mcu = true;
120        }
121        for cell in [
122            "AND2_",
123            "INV ",
124            "BUF ",
125            "DFF ",
126            "DFFSR ",
127            "LATCH ",
128            "CKLNQD ",
129            "GEM_ASSERT ",
130            "GEM_DISPLAY ",
131            "$__RAMGEM_",
132        ] {
133            if line.contains(cell) {
134                has_aigpdk = true;
135                break;
136            }
137        }
138        if (has_aigpdk as u8 + has_sky130 as u8 + has_gf180mcu as u8) >= 2 {
139            return Ok(CellLibrary::Mixed);
140        }
141    }
142
143    Ok(resolve_library(has_aigpdk, has_sky130, has_gf180mcu))
144}

146// ============================================================================
147// PdkVariant — dispatch enum for standard-cell PDK predicates / pin names.
148// ============================================================================
149
150/// Standard-cell PDK identification, used by `aig.rs` to dispatch
151/// per-cell preprocessing and post-processing without `if/else if`
152/// ladders on `is_sky130_cell` / `is_gf180mcu_cell`.
153///
154/// Variants cover only the standard-cell PDKs that own behavioural
155/// models loaded from vendored submodules. AIGPDK (the internal cell
156/// library) is handled inline in `aig.rs` because its cells are
157/// hard-coded and don't go through the `decompose_with_pdk` path.
158#[derive(Debug, Clone, Copy, PartialEq, Eq)]
159pub enum PdkVariant {
160    /// google/skywater-pdk-libs-sky130_fd_sc_hd
161    Sky130,
162    /// gf180mcu_fd_sc_mcu7t5v0 + mcu9t5v0
163    Gf180Mcu,
164}
165
166impl PdkVariant {
167    /// Identify the PDK variant a vendor-prefixed cell type belongs to.
168    /// Returns `None` if the cell isn't a recognised standard-cell PDK
169    /// instance (i.e. it's AIGPDK, an SRAM macro, or unknown).
170    pub fn classify(celltype: &str) -> Option<PdkVariant> {
171        if is_sky130_cell(celltype) {
172            Some(PdkVariant::Sky130)
173        } else if is_gf180mcu_cell(celltype) {
174            Some(PdkVariant::Gf180Mcu)
175        } else {
176            None
177        }
178    }
179
180    /// Strip the vendor prefix from `celltype` to get the base cell
181    /// type (e.g. `sky130_fd_sc_hd__inv_2` → `inv`,
182    /// `gf180mcu_fd_sc_mcu7t5v0__nand2_4` → `nand2`).
183    pub fn extract_cell_type<'a>(self, celltype: &'a str) -> &'a str {
184        match self {
185            PdkVariant::Sky130 => crate::sky130::extract_cell_type(celltype),
186            PdkVariant::Gf180Mcu => crate::gf180mcu::extract_cell_type(celltype),
187        }
188    }
189
190    /// True if the base cell type is a sequential element (DFF / latch /
191    /// clock-gating cell).
192    pub fn is_sequential(self, cell_type: &str) -> bool {
193        match self {
194            PdkVariant::Sky130 => crate::sky130_pdk::is_sequential_cell(cell_type),
195            PdkVariant::Gf180Mcu => crate::gf180mcu_pdk::is_sequential_cell(cell_type),
196        }
197    }
198
199    /// True if the base cell type is a constant-driving tie cell.
200    pub fn is_tie(self, cell_type: &str) -> bool {
201        match self {
202            PdkVariant::Sky130 => crate::sky130_pdk::is_tie_cell(cell_type),
203            PdkVariant::Gf180Mcu => crate::gf180mcu_pdk::is_tie_cell(cell_type),
204        }
205    }
206
207    /// True if the cell is a physical-only filler / decap / antenna cell
208    /// that contributes no logic to the AIG. SKY130 has no filler concept
209    /// (its fillers are handled differently); this returns `false` for
210    /// SKY130.
211    pub fn is_filler(self, cell_type: &str) -> bool {
212        match self {
213            PdkVariant::Sky130 => false,
214            PdkVariant::Gf180Mcu => crate::gf180mcu_pdk::is_filler_cell(cell_type),
215        }
216    }
217
218    /// True if the cell has multiple functional outputs (full / half
219    /// adders, dual-output DFFs).
220    pub fn is_multi_output(self, cell_type: &str) -> bool {
221        match self {
222            PdkVariant::Sky130 => crate::sky130_pdk::is_multi_output_cell(cell_type),
223            PdkVariant::Gf180Mcu => crate::gf180mcu_pdk::is_multi_output_cell(cell_type),
224        }
225    }
226
227    /// True if the cell is a logic-bearing IO pad (input / inout buffer).
228    /// SKY130 IO pads are not currently supported through the digital
229    /// path; this returns `false` for SKY130.
230    pub fn is_io_pad(self, cell_type: &str) -> bool {
231        match self {
232            PdkVariant::Sky130 => false,
233            PdkVariant::Gf180Mcu => crate::gf180mcu_pdk::is_io_pad_cell(cell_type),
234        }
235    }
236
237    /// Pin name for the active-low asynchronous reset on a sequential
238    /// cell. SKY130: `RESET_B`; GF180MCU: `RN`.
239    pub fn reset_pin(self) -> &'static str {
240        match self {
241            PdkVariant::Sky130 => "RESET_B",
242            PdkVariant::Gf180Mcu => "RN",
243        }
244    }
245
246    /// Pin name for the active-low asynchronous set on a sequential
247    /// cell. SKY130: `SET_B`; GF180MCU: `SETN`.
248    pub fn set_pin(self) -> &'static str {
249        match self {
250            PdkVariant::Sky130 => "SET_B",
251            PdkVariant::Gf180Mcu => "SETN",
252        }
253    }
254}

256#[cfg(test)]
257mod tests {
258    use super::*;
259
260    #[test]
261    fn test_detect_library() {
262        // SKY130 only
263        let sky130_cells = ["sky130_fd_sc_hd__inv_2", "sky130_fd_sc_hd__nand2_1"];
264        assert_eq!(detect_library(sky130_cells.iter()), CellLibrary::SKY130);
265
266        // AIGPDK only
267        let aigpdk_cells = ["AND2_00_0", "INV", "DFF"];
268        assert_eq!(detect_library(aigpdk_cells.iter()), CellLibrary::AIGPDK);
269
270        // GF180MCU only — 7t + 9t coexist within one library.
271        let gf180mcu_cells = [
272            "gf180mcu_fd_sc_mcu7t5v0__inv_2",
273            "gf180mcu_fd_sc_mcu9t5v0__nand2_1",
274        ];
275        assert_eq!(detect_library(gf180mcu_cells.iter()), CellLibrary::GF180MCU);
276
277        // Mixed — any two libraries together is an error.
278        let mixed_sky_aig = ["sky130_fd_sc_hd__inv_2", "AND2_00_0"];
279        assert_eq!(detect_library(mixed_sky_aig.iter()), CellLibrary::Mixed);
280        let mixed_gf_sky = [
281            "gf180mcu_fd_sc_mcu7t5v0__inv_2",
282            "sky130_fd_sc_hd__nand2_1",
283        ];
284        assert_eq!(detect_library(mixed_gf_sky.iter()), CellLibrary::Mixed);
285        let mixed_gf_aig = ["gf180mcu_fd_sc_mcu7t5v0__inv_2", "AND2_00_0"];
286        assert_eq!(detect_library(mixed_gf_aig.iter()), CellLibrary::Mixed);
287
288        // Empty defaults to AIGPDK
289        let empty: Vec<&str> = vec![];
290        assert_eq!(detect_library(empty.iter()), CellLibrary::AIGPDK);
291    }
292
293    #[test]
294    fn pdk_variant_classify() {
295        assert_eq!(
296            PdkVariant::classify("sky130_fd_sc_hd__inv_2"),
297            Some(PdkVariant::Sky130)
298        );
299        assert_eq!(
300            PdkVariant::classify("gf180mcu_fd_sc_mcu7t5v0__nand2_4"),
301            Some(PdkVariant::Gf180Mcu)
302        );
303        assert_eq!(PdkVariant::classify("AND2_00_0"), None);
304        assert_eq!(PdkVariant::classify("DFF"), None);
305    }
306
307    #[test]
308    fn pdk_variant_async_pin_names() {
309        assert_eq!(PdkVariant::Sky130.reset_pin(), "RESET_B");
310        assert_eq!(PdkVariant::Sky130.set_pin(), "SET_B");
311        assert_eq!(PdkVariant::Gf180Mcu.reset_pin(), "RN");
312        assert_eq!(PdkVariant::Gf180Mcu.set_pin(), "SETN");
313    }
