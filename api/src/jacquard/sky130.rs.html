3//! SKY130 standard cell library support for GEM.
4//!
5//! This module provides pin direction and width information for the SKY130
6//! open-source PDK standard cells. SKY130 cells follow the naming convention:
7//! `sky130_fd_sc_<variant>__<celltype>_<drive>`
8//!
9//! Supported variants: hd (high-density), hs (high-speed), ms (medium-speed)
10
11use compact_str::CompactString;
12use netlistdb::{Direction, LeafPinProvider};
13use sverilogparse::SVerilogRange;
14
15// Generated by build.rs from vendor/sky130_fd_sc_hd/cells. Provides
16// `SKY130_PIN_TABLE: &[(&str, &[(&str, Direction)])]` — the
17// authoritative pin-direction table for every SKY130 standard cell
18// (drive-strength suffix already stripped). Consumed below by
19// `SKY130LeafPins::direction_of` via exact lookup, mirroring the
20// gf180mcu pattern. The previous hand-rolled prefix-match arms were
21// missing entries for ~42 cell families (DFFs with `Q_N`, neg-edge
22// DFFs, set+reset DFFs, scan / DE / latch variants, tristate buffers,
23// `fa` variants with named SUM/COUT, low-power isolation, probe /
24// spare cells) — all of which the build.rs-generated table covers
25// correctly. The shadow bug where `t.starts_with("buf")` swallowed
26// `bufinv_*` (panic on `Y`) likewise disappears with exact lookup.
27mod generated {
28    use netlistdb::Direction;
29    include!(concat!(env!("OUT_DIR"), "/sky130_pins.rs"));
30}
31
32use generated::SKY130_PIN_TABLE;

34/// LeafPinProvider implementation for SKY130 standard cells.
35///
36/// Standard-cell pin directions come from the build.rs-generated
37/// `SKY130_PIN_TABLE`. The ChipFlow `CF_SRAM_*` macros and the bare
38/// AIGPDK `INV` cell (which sometimes appears inline in SKY130
39/// netlists synthesized from `assign ~expr`) are handled out of band
40/// because they are not part of the vendored cell catalogue.
41pub struct SKY130LeafPins;
42
43impl LeafPinProvider for SKY130LeafPins {
44    fn direction_of(
45        &self,
46        macro_name: &CompactString,
47        pin_name: &CompactString,
48        _pin_idx: Option<isize>,
49    ) -> Direction {
50        let cell_type = extract_cell_type(macro_name);
51        let pin = pin_name.as_str();
52
53        // AIGPDK INV (synthesized from `assign ~expr`); not in the
54        // SKY130 cell catalogue but can appear inline in mixed
55        // netlists. `extract_cell_type("INV")` returns "INV" verbatim
56        // because there is no library prefix to strip.
57        if cell_type == "INV" {
58            return match pin {
59                "A" => Direction::I,
60                "Y" => Direction::O,
61                _ => unknown_pin(macro_name, pin_name),
62            };
63        }
64
65        // Standard cells: exact lookup against the generated table.
66        for (ct, pins) in SKY130_PIN_TABLE {
67            if *ct == cell_type {
68                for (p, d) in *pins {
69                    if *p == pin {
70                        return *d;
71                    }
72                }
73                panic!(
74                    "Unknown SKY130 pin on cell type '{cell_type}': \
75                     macro={macro_name}, pin={pin_name}",
76                );
77            }
78        }
79
80        // ChipFlow SRAM macros (`CF_SRAM_*`) — not in the SKY130 cell
81        // catalogue. Match on the full macro name (no prefix-stripping
82        // applies here).
83        if macro_name.starts_with("CF_SRAM_") {
84            return match pin {
85                // Control inputs
86                "WLBI" | "WLOFF" | "CLKin" | "EN" | "R_WB" | "SM" | "TM" => Direction::I,
87                "ScanInDR" | "ScanInDL" | "ScanInCC" => Direction::I,
88                "vpwrpc" | "vpwrac" => Direction::I,
89                // Data/address inputs
90                "AD" | "BEN" | "DI" => Direction::I,
91                // Data/scan outputs
92                "DO" | "ScanOutCC" => Direction::O,
93                _ => unknown_pin(macro_name, pin_name),
94            };
95        }
96
97        panic!(
98            "Unknown SKY130 cell (no pin-table entry for type '{cell_type}'): \
99             macro={macro_name}",
100        );
101    }
102
103    fn width_of(
104        &self,
105        macro_name: &CompactString,
106        pin_name: &CompactString,
107    ) -> Option<SVerilogRange> {
108        // Check for ChipFlow SRAM macros which have bus pins
109        if macro_name.starts_with("CF_SRAM_") {
110            match pin_name.as_str() {
111                // Data buses are 32-bit
112                "DO" | "DI" | "BEN" => Some(SVerilogRange(31, 0)),
113                // Address bus - width depends on SRAM depth
114                // CF_SRAM_1024x32 has 10-bit address (2^10 = 1024)
115                "AD" => Some(SVerilogRange(9, 0)),
116                // All other pins are single-bit
117                _ => None,
118            }
119        } else {
120            // All SKY130 standard cells have single-bit pins
121            None
122        }
123    }
124}

126/// Extract the cell type from a SKY130 macro name.
127///
128/// Converts `sky130_fd_sc_hd__inv_2` to `inv`.
129pub fn extract_cell_type(name: &str) -> &str {
130    // First strip the library prefix
131    let stripped = name
132        .strip_prefix("sky130_fd_sc_hd__")
133        .or_else(|| name.strip_prefix("sky130_fd_sc_hs__"))
134        .or_else(|| name.strip_prefix("sky130_fd_sc_ms__"))
135        .or_else(|| name.strip_prefix("sky130_fd_sc_ls__"))
136        .or_else(|| name.strip_prefix("sky130_fd_sc_lp__"))
137        .or_else(|| name.strip_prefix("sky130_fd_sc_hdll__"))
138        .or_else(|| name.strip_prefix("sky130_fd_sc_hvl__"))
139        .unwrap_or(name);
140
141    // Then strip the drive strength suffix (e.g., _2, _4, _16)
142    // The drive strength is always a number at the end after an underscore
143    if let Some(last_underscore) = stripped.rfind('_') {
144        let suffix = &stripped[last_underscore + 1..];
145        // Check if suffix is purely numeric (drive strength)
146        if !suffix.is_empty() && suffix.chars().all(|c| c.is_ascii_digit()) {
147            return &stripped[..last_underscore];
148        }
149    }
150
151    stripped
152}

154/// Return Direction::I as fallback but log an error.
155fn unknown_pin(macro_name: &CompactString, pin_name: &CompactString) -> Direction {
156    panic!("Unknown SKY130 pin: macro={}, pin={}", macro_name, pin_name);
157}

159/// Check if a cell name is a SKY130 standard cell or associated macro.
160pub fn is_sky130_cell(name: &str) -> bool {
161    name.starts_with("sky130_fd_sc_") || name.starts_with("CF_SRAM_")
162}
163
164/// Check if a cell name is an AIGPDK cell.
165pub fn is_aigpdk_cell(name: &str) -> bool {
166    matches!(
167        name,
168        "INV"
169            | "BUF"
170            | "CKLNQD"
171            | "AND2_00_0"
172            | "AND2_01_0"
173            | "AND2_10_0"
174            | "AND2_11_0"
175            | "AND2_11_1"
176            | "DFF"
177            | "DFFSR"
178            | "LATCH"
179            | "GEM_ASSERT"
180            | "GEM_DISPLAY"
181    ) || name.starts_with("$__RAMGEM_")
182}
